120 static void RemoveAnalogInputByID(uint8_t id) {
121  if (isExternalInput(id)) {
122  InitializeInput(&Ext_AInputs[id]);
123  } else if (isInternalInput(id)) {
124  if (id == IN_COLD_JUNCTION) {
125  /* We don't want to allow removal of this, so we return immediately. */
126  return;
127  }
128  InitializeInput(&Int_AInputs[id - (NUM_EXT_ANALOG_INPUTS + NUM_CAL_ANALOG_INPUTS)]);
129  } else if (id == EXTERNAL_OFFSET_CAL) {
130  /* Do nothing, we don't want to remove this input. */
131  } else {
132  /* This is out of range */
133 #ifdef ANALOGINPUT_DEBUG
134  printf("[Analog Input] Cannot find the requested analog input. Input does not exist on the board.\n\r");
135 #endif
136  }
137 }

150 void AnalogInputsInit(void) {
151  /* Initialize the input multiplexer */
152  InputMultiplexerInit();
153  uint_fast8_t i = 0;
154  for (; i < NUM_EXT_ANALOG_INPUTS; ++i) {
155  InitializeInput(&Ext_AInputs[i]);
156  }
157  for (i = 0; i < NUM_INT_ANALOG_INPUTS; ++i) {
158  InitializeInput(&Int_AInputs[i]);
159  }
160  /* Configure the offset cal input */
161  InitializeInput(&Offset_Cal_AInput);
162 
163  /* Configure the cold junction input */
164  Analog_Input_t* cold = GetAnalogInputByNumber(IN_COLD_JUNCTION);
165  cold->physicalInput = IN_COLD_JUNCTION;
166  cold->buffer = ADS1256_BUFFER_ENABLED;
167  cold->rate = ADS1256_SPS_30000;
168  cold->gain = ADS1256_PGAx8;
169  strcpy(cold->name, "COLD JUNCTION");
170  cold->bufferReadIdx = 0U;
171  cold->bufferWriteIdx = 0U;
172  cold->min = 0;
173  cold->max = 0;
174  AddAnalogInput(cold);

267 Tekdaqc_Function_Error_t CreateAnalogInput(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
268  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
269  char* param;
270  int8_t index = -1;
271  /* Set the default parameters */
272  uint8_t input = NULL_CHANNEL; /* The physical input */
273  ADS1256_BUFFER_t buffer = ADS1256_BUFFER_ENABLED; /* The default buffer setting */
274  ADS1256_SPS_t rate = ADS1256_SPS_10; /* The default sample rate setting */
275  ADS1256_PGA_t gain = ADS1256_PGAx1; /* The default gain setting */
276  char name[MAX_ANALOG_INPUT_NAME_LENGTH]; /* The name */
277  strcpy(name, "NONE");
278  uint_fast8_t i = 0U;
279  for (; i < NUM_ADD_ANALOG_INPUT_PARAMS; ++i) {
280  index = GetIndexOfArgument(keys, ADD_ANALOG_INPUT_PARAMS[i], count);
281  if (index >= 0) { /* We found the key in the list */
282  param = values[index]; /* We use the discovered index for this key */
283  switch (i) { /* Switch on the key not position in arguments list */
284  case 0U: { /* INPUT key */
285  char* testPtr = NULL;
286  uint8_t in = (uint8_t) strtol(param, &testPtr, 10);
287  if (testPtr == param) {
288  retval = ERR_AIN_PARSE_ERROR;
289  } else {
290  if (in >= 0U && in <= NUM_ANALOG_INPUTS) {
291  /* A valid input number */
292  input = in;
293  } else {
294  /* Input number out of range */
295 #ifdef ANALOGINPUT_DEBUG
296  printf("[Analog Input] The requested input number is invalid.\n\r");
297 #endif
298  retval = ERR_AIN_INPUT_OUTOFRANGE;
299  }
300  }
301  break;
302  }
303  case 1U: /* BUFFER key */
304  buffer = ADS1256_StringToBuffer(param);
305  break;
306  case 2U: /* RATE key */
307  rate = ADS1256_StringToDataRate(param);
308  break;
309  case 3U: /* GAIN key */
310  gain = ADS1256_StringToPGA(param);
311  break;
312  case 4U: /* NAME key */
313  strcpy(name, param);
314  break;
315  default:
316  retval = ERR_AIN_PARSE_ERROR;
317  }
318  } else if (i == 1U || i == 2U || i == 3U || i == 4U || i == 5U) {
319  /* The BUFFER, RATE, GAIN and NAME keys are not strictly required, leave the defaults */
320  continue;
321  } else {
322  /* Somehow an error happened */
323 #ifdef ANALOGINPUT_DEBUG
324  printf("[Analog Input] Unable to locate required key: %s\n\r", ADD_ANALOG_INPUT_PARAMS[i]);
325 #endif
326  retval = ERR_AIN_PARSE_MISSING_KEY; /* Failed to locate a key */
327  }
328  if (retval != ERR_FUNCTION_OK) {
329  break;
330  }
331  }
332  if (retval == ERR_FUNCTION_OK) {
333  if (input != NULL_CHANNEL ) {
334  Analog_Input_t* an_input = GetAnalogInputByNumber(input);
335  if (an_input != NULL ) {
336  if (an_input->added == CHANNEL_NOTADDED) {
337  an_input->physicalInput = input;
338  an_input->buffer = buffer;
339  an_input->rate = rate;
340  an_input->gain = gain;
341  strcpy(an_input->name, name);
342  an_input->bufferReadIdx = 0U;
343  an_input->bufferWriteIdx = 0U;
344  an_input->min = 0;
345  an_input->max = 0;
346  retval = AddAnalogInput(an_input);
347  } else {
348  retval = ERR_AIN_INPUT_EXISTS;
349  }
350  } else {
351  /* The input could not be found */
352  retval = ERR_AIN_INPUT_NOT_FOUND;
353  }
354  } else {
355  /* A valid input was never specified */
356  retval = ERR_AIN_INPUT_UNSPECIFIED;
357  }
358  }
359  return retval; /* Return the status */
360 }

368 Tekdaqc_Function_Error_t AddAnalogInput(Analog_Input_t* input) {
369  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
370  PhysicalAnalogInput_t index = input->physicalInput;
371  if (isExternalInput(index)) {
372  /* This is an external analog input */
373  input->externalInput = GetExternalMuxedInputByNumber(index);
374  input->internalInput = NULL_CHANNEL;
375  input->added = CHANNEL_ADDED;
376 #ifdef ANALOGINPUT_DEBUG
377  printf("[Analog Input] Added input to external inputs list.\n\r");
378 #endif
379  } else if (isInternalInput(index)) {
380  /* This is an internal analog input */
381  input->externalInput = NULL_CHANNEL;
382  input->internalInput = index - (NUM_EXT_ANALOG_INPUTS + NUM_CAL_ANALOG_INPUTS);
383  input->added = CHANNEL_ADDED;
384 #ifdef ANALOGINPUT_DEBUG
385  printf("[Analog Input] Added input to internal inputs list.\n\r");
386 #endif
387  } else if (index == EXTERNAL_OFFSET_CAL) {
388  /* This is the offset calibration input */
389  input->externalInput = NULL_CHANNEL;
390  input->internalInput = NULL_CHANNEL;
391  input->added = CHANNEL_ADDED;
392 #ifdef ANALOGINPUT_DEBUG
393  printf("[Analog Input] Added external offset input.\n\r");
394 #endif
395  } else {
396  /* This is out of range */
397 #ifdef ANALOGINPUT_DEBUG
398  printf("[Analog Input] Cannot add the provided analog input. Input does not exist on the board.\n\r");
399 #endif
400  retval = ERR_AIN_INPUT_OUTOFRANGE;
401  }
402  return retval;
403 }

480 void WriteAnalogInput(Analog_Input_t* input) {
481  uint8_t count = 0;
482  uint8_t retval;
483  while (count < SINGLE_ANALOG_WRITE_COUNT && input->bufferReadIdx != input->bufferWriteIdx) {
484  /* We have data to print */
485  if (count == 0) {
486  retval = sprintf(TOSTRING_BUFFER, ANALOG_INPUT_HEADER, input->name, input->physicalInput, ADS1256_StringFromPGA(input->gain),
487  ADS1256_StringFromSPS(input->rate), ADS1256_StringFromBuffer(input->buffer));
488  writer(TOSTRING_BUFFER);
489  }
490  retval = sprintf(TOSTRING_BUFFER, "%" PRIu64 ", %" PRIi32 "\n\r", input->timestamps[input->bufferReadIdx],
491  input->values[input->bufferReadIdx]);
492  input->bufferReadIdx = (input->bufferReadIdx + 1) % ANALOG_INPUT_BUFFER_SIZE;
493  if (retval >= 0) {
494  if (writer != 0) {
495  writer(TOSTRING_BUFFER);
496  } else {
497 #ifdef ANALOGINPUT_DEBUG
498  printf("[Analog Input] Cannot write analog input to string due to NULL write function.\n\r");
499 #endif
500  }
501  } else {
502 #ifdef ANALOGINPUT_DEBUG
503  printf("[Analog Input] Error occurred while writing analog input to string.\n\r");
504 #endif
505  }
506  ++count;
507  }
508  if (writer != 0) {
509  writer("\x1E");
510  }
511 }
