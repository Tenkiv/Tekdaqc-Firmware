144 void DO_Machine_Service(void) {
145  /* Determine the state */
146  switch (state) {
147  case DO_UNINITIALIZED:
148  DO_Machine_Init();
149  break;
150  case DO_INITIALIZED:
151  DO_Machine_Idle();
152  break;
153  case DO_IDLE:
154  break;
155  case DO_CHANNEL_SAMPLING: {
156  Digital_Output_t* output = NULL;
157  if (SampleCurrent < SampleTotal) {
158  if (numberSamplingOutputs == 1) {
159  /* TODO: Output channel sampling */
160  ++SampleCurrent;
161  } else {
162  for (uint_fast8_t i = 0; i < NUM_DIGITAL_OUTPUTS; ++i) {
163  output = samplingOutputs[i];
164  if (output != NULL) {
165  /* TODO: Output channel sampling */
166  }
167  }
168  ++SampleCurrent;
169  }
170  } else {
171 #ifdef DO_STATE_MACHINE_DEBUG
172  printf("[DO STATE MACHINE] Channel sampling is complete.\n\r");
173 #endif
174  TelnetWriteStatusMessage("DO Channel sampling completed.");
175  DO_Machine_Idle();
176  CompletedDOSampling();
177  }
178  }
179  break;
180  case DO_RESET:
181  SampleCurrent = 0U;
182  SampleTotal = 0U;
183  samplingOutputs = NULL;
184  numberSamplingOutputs = 0U;
185  DO_Machine_Idle();
186  break;
187  default:
188  /* TODO: Throw an error */
189  break;
190  }
191 }

241 void DO_Machine_Output_Sample(Digital_Output_t** outputs, uint32_t count, bool singleChannel) {
242  /* We use explicit checking here in case other state are added later */
243  if ((state != DO_IDLE)) {
244 #ifdef DO_STATE_MACHINE_DEBUG
245  printf("[DO STATE MACHINE] Attempted to enter DO_CHANNEL_SAMPLING state from %s\n\r", DOMachine_StringFromState(state));
246 #endif
247  return;
248  }
249 
250  /* Validate the outputs list */
251  if (outputs == NULL) {
252 #ifdef DO_STATE_MACHINE_DEBUG
253  printf("[DO STATE MACHINE] Attempted to enter DO_CHANNEL_SAMPLING state with a NULL digital output list. Ignoring...\n\r");
254 #endif
255  return;
256  }
257  if (singleChannel == true) {
258  /* Validate the output(s) */
259  if (outputs[0] == NULL) {
260 #ifdef DO_STATE_MACHINE_DEBUG
261  printf("[DO STATE MACHINE] Attempted to enter DO_CHANNEL_SAMPLING state with a NULL digital output. Ignoring...\n\r");
262 #endif
263  return;
264  }
265  /* Select output */
266  currentSamplingOutput = 0; /* Use i here so we don't loose any work from the search earlier */
267  numberSamplingOutputs = 1;
268  } else {
269  /* Validate the output(s) */
270  uint8_t i = 0U;
271  for (; i < NUM_DIGITAL_OUTPUTS; ++i) {
272  if (outputs[i] != NULL) {
273  break;
274  }
275  }
276  if (i >= NUM_DIGITAL_OUTPUTS) {
277 #ifdef DO_STATE_MACHINE_DEBUG
278  printf("[DO STATE MACHINE] Attempted to enter DO_CHANNEL_SAMPLING state with NULL digital outputs. Ignoring...\n\r");
279 #endif
280  return;
281  }
282  /* Select output */
283  currentSamplingOutput = i; /* Use i here so we don't loose any work from the search earlier */
284  numberSamplingOutputs = NUM_DIGITAL_OUTPUTS;
285  }
286 
287  /* Select output */
288  samplingOutputs = outputs;
289  uint8_t i = 0U;
290  Digital_Output_t* output = samplingOutputs[currentSamplingOutput];
291  while (output == NULL) { /* Keep searching until we find a non-null output */
292  output = samplingOutputs[++i];
293  if (i == NUM_DIGITAL_OUTPUTS) {
294  /* We have reached the end of a set */
295  /* There is nothing to sample */
296  return;
297  }
298  }
299  /* Save current time and sample count */
300  SampleCurrent = 0U;
301  SampleTotal = count;
302  state = DO_CHANNEL_SAMPLING;
303 }
