81 static void SelectExternalInput(ExternalMuxedInput_t input, bool doMuxDelay) {
82  /* Verify that the internal input parameter is set */
83  if (input == NULL_CHANNEL ) {
84 #ifdef INPUT_MULTIPLEXER_DEBUG
85  printf("[Analog Input Multiplexer] Attempted to select an input as external when it is not configured to be.\n\r");
86 #endif
87  return;
88  }
89  /* Make the switch */
90  SelectInternalInput(EXTERNAL_ANALOG_IN);
91  GPIO_Write(EXT_ANALOG_IN_MUX_PORT, (input | (GPIO_ReadOutputData(EXT_ANALOG_IN_MUX_PORT ) & EXT_ANALOG_IN_BITMASK )));
92 
93  if (doMuxDelay == true) {
94  /* Wait for the external multiplexing relays to conduct */
95  MuxCompleteTime = GetLocalTime() + EXTERNAL_MUX_DELAY;
96  /* Change ADC state machine to ADC_MUXING state */
97  ADC_External_Muxing();
98  }
99 }

108 static void SelectInternalInput(InternalAnalogInput_t input) {
109  /* Verify that the internal input parameter is set */
110  if (input == NULL_CHANNEL ) {
111 #ifdef INPUT_MULTIPLEXER_DEBUG
112  printf("[Analog Input Multiplexer] Attempted to select an input as internal when it is not configured to be.\n\r");
113 #endif
114  return;
115  }
116  ADS1256_AIN_t pos;
117  ADS1256_AIN_t neg;
118  /* Make the switch */
119  switch (input) {
120  case SUPPLY_9V:
121  pos = SUPPLY_9V_AINP;
122  neg = SUPPLY_9V_AINN;
123  break;
124  case SUPPLY_5V:
125  pos = SUPPLY_5V_AINP;
126  neg = SUPPLY_5V_AINN;
127  break;
128  case SUPPLY_3_3V:
129  pos = SUPPLY_3_3V_AINP;
130  neg = SUPPLY_3_3V_AINN;
131  break;
132  case COLD_JUNCTION:
133  pos = COLD_JUNCTION_AINP;
134  neg = COLD_JUNCTION_AINN;
135  break;
136  case EXTERNAL_ANALOG_IN:
137  pos = EXTERNAL_ANALOG_IN_AINP;
138  neg = EXTERNAL_ANALOG_IN_AINN;
139  break;
140  default:
141 #ifdef INPUT_MULTIPLEXER_DEBUG
142  printf("[Analog Input Multiplexer] The requested internal input is invalid.\n\r");
143 #endif
144  return;
145  }
146  ADS1256_SetInputChannels(pos, neg);
147 }

197 void InputMultiplexerInit(void) {
198  GPIO_InitTypeDef GPIO_InitStructure;
199 
200  /* Enable the GPIO Clock */
201  RCC_AHB1PeriphClockCmd(OCAL_CONTROL_GPIO_CLK, ENABLE);
202 
203  /* Configure the DEMUX GPIO pins */
204  GPIO_InitStructure.GPIO_Pin = OCAL_CONTROL_PIN;
205  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
206  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
207  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
208  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz; /* At most the multiplexer can handle a few hundred Hz */
209  GPIO_Init(OCAL_CONTROL_GPIO_PORT, &GPIO_InitStructure);
210  GPIO_WriteBit(OCAL_CONTROL_GPIO_PORT, OCAL_CONTROL_PIN, OCAL_SELECT); /* TODO: This is temporary until the REV D boards arrive. Should be EXT_ANALOG_SELECT */
211 
212  /* Enable the DEMUX GPIO Clock */
213  RCC_AHB1PeriphClockCmd(EXT_ANALOG_IN_GPIO_CLK, ENABLE);
214 
215  /* Configure the DEMUX GPIO pins */
216  GPIO_InitStructure.GPIO_Pin = EXT_ANALOG_IN_MUX_PINS;
217  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
218  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
219  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
220  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz; /* At most the multiplexer can handle a few hundred Hz */
221  GPIO_Init(EXT_ANALOG_IN_MUX_PORT, &GPIO_InitStructure);
222  GPIO_Write(EXT_ANALOG_IN_MUX_PORT, (EXTERN_0 | (GPIO_ReadOutputData(EXT_ANALOG_IN_MUX_PORT ) & EXT_ANALOG_IN_BITMASK )));
223 }

232 void SelectAnalogInput(Analog_Input_t* input, bool doMuxDelay) {
233  if (input != NULL ) {
234  if (isExternalInput(input->physicalInput)) {
235  SelectExternalInput(input->externalInput, doMuxDelay);
236  CurrentInput = input;
237  } else if (isInternalInput(input->physicalInput)) {
238  SelectInternalInput(input->internalInput);
239  CurrentInput = input;
240  } else if (input->physicalInput == EXTERNAL_OFFSET_CAL) {
241  SelectCalibrationInput();
242  CurrentInput = input;
243  } else {
244  /* ERROR */
245  snprintf(TOSTRING_BUFFER, sizeof(TOSTRING_BUFFER),
246  "[Analog Input Multiplexer] Attempted to select an input which does not exist.");
247  TelnetWriteErrorMessage(TOSTRING_BUFFER);
248 #ifdef INPUT_MULTIPLEXER_DEBUG
249  printf("[Analog Input Multiplexer] Attempted to select an input which does not exist.\n\r");
250 #endif
251  }
252  } else {
253 #ifdef INPUT_MULTIPLEXER_DEBUG
254  printf("[Analog Input Multiplexer] Attempted to select a NULL input.\n\r");
255 #endif
256  }
257 }
