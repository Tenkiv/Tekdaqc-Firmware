137 static void program_loop(void) {
138  /* Infinite loop */
139  while (1) {
140  /* Service the inputs/outputs */
141  ServiceTasks();
142 
143  /* Check if any packet received */
144  if (ETH_CheckFrameReceived()) {
145  /* Process received ethernet packet */
146  LwIP_Pkt_Handle();
147  }
148 
149  /* Handle periodic timers for LwIP */
150  LwIP_Periodic_Handle(GetLocalTime());
151 
152  if (TelnetIsConnected() == true) { /* We have an active Telnet connection to service */
153  /* Do server stuff */
154  character = TelnetRead();
155  if (character != '\0') {
156  Command_AddChar(character);
157  }
158  }
159 
160  /* Check to see if any faults have occurred */
161  Tekdaqc_CheckStatus();
162 
163  /* Reload the IWDG Counter to prevent reset */
164  IWDG_ReloadCounter();
165  }
166 }

204  DigitalOutputsInit();
205 
206  /* Set the write functions */
207  SetAnalogInputWriteFunction(&TelnetWriteString);
208  SetDigitalInputWriteFunction(&TelnetWriteString);
209 
210  /* Initialize the FLASH disk */
211  FlashDiskInit();
212 
213  /* Initialize the calibration table and fetch the board serial number */
214  Tekdaqc_CalibrationInit();
215  char data = '\0';
216  uint32_t Address = BOARD_SERIAL_NUM_ADDR;
217  for (int i = 0; i < BOARD_SERIAL_NUM_LENGTH; ++i) {
218  if (Address >= CAL_TEMP_LOW_ADDR) {
219  /* We have overflowed and there is a problem in the software */
220 #ifdef DEBUG
221  printf("[Config] Reading board serial number overflowed.\n\r");
222 #endif
223  break; /* Break out to prevent an invalid access */
224  }
225  data = *((__IO char*)Address);
226  TEKDAQC_BOARD_SERIAL_NUM[i] = data;
227  Address += sizeof(char);
228  }
229  TEKDAQC_BOARD_SERIAL_NUM[BOARD_SERIAL_NUM_LENGTH] = '\0'; /* Apply the NULL termination character */
230 
231 #ifdef USE_WATCHDOG
232  // Initialize the watchdog timer
233  Watchdog_Init();
234 #endif
235 }
