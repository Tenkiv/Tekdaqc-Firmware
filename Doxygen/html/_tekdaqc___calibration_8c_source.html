67 static Tekdaqc_Function_Error_t SetADCParameters(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
68  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
69  int8_t index = -1;
70  ADS1256_PGA_t pga = ADS1256_PGAx1;
71  ADS1256_SPS_t rate = ADS1256_SPS_60;
72  ADS1256_BUFFER_t buffer = ADS1256_BUFFER_DISABLED;
73  for (uint_fast8_t i = 0; i < NUM_SYSTEM_CAL_PARAMS; ++i) {
74  index = GetIndexOfArgument(keys, SYSTEM_CAL_PARAMS[i], count);
75  if (index >= 0) { /* We found the key in the list */
76  switch (i) { /* Switch on the key not position in arguments list */
77  case 0: /* BUFFER key */
78  buffer = ADS1256_StringToBuffer(values[index]);
79  break;
80  case 1: /* RATE key */
81  rate = ADS1256_StringToDataRate(values[index]); /* We use the discovered index for this key */
82  break;
83  case 2: /* GAIN key */
84  pga = ADS1256_StringToPGA(values[index]); /* We use the discovered index for this key */
85  break;
86  default:
87  retval = ERR_AIN_PARSE_ERROR;
88  }
89  } else if (i == 0 || i == 1 || i == 2 || i == 3) {
90  /* The GAIN, RATE and BUFFER keys are not strictly required */
91  continue;
92  } else {
93  /* Somehow an error happened */
94 #ifdef CALIBRATION_DEBUG
95  printf("[Command Interpreter] Unable to locate key: %s\n\r", READ_ANALOG_KEYS[i]);
96 #endif
97  retval = ERR_AIN_PARSE_MISSING_KEY; /* Failed to locate a key */
98  }
99  }
100  ADS1256_SetInputBufferSetting(buffer);
101  ADS1256_SetDataRate(rate);
102  ADS1256_SetPGASetting(pga);
103  return retval;
104 }

132 Tekdaqc_Function_Error_t PerformSystemGainCalibration(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
133  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
134  retval = SetADCParameters(keys, values, count);
135  PhysicalAnalogInput_t input = EXTERNAL_0;
136  int8_t index = -1;
137  for (uint_fast8_t i = 0; i < NUM_SYSTEM_GCAL_PARAMS; ++i) {
138  index = GetIndexOfArgument(keys, SYSTEM_GCAL_PARAMS[i], count);
139  if (index >= 0) { /* We found the key in the list */
140  switch (i) { /* Switch on the key not position in arguments list */
141  case 3: /* INPUT key */
142  input = ADS1256_StringToBuffer(values[index]);
143  break;
144  default:
145  retval = ERR_AIN_PARSE_ERROR;
146  }
147  } else if (i == 0 || i == 1 || i == 2) {
148  /* The GAIN, RATE and BUFFER keys are not strictly required */

150  } else {
151  /* Somehow an error happened */
152 #ifdef CALIBRATION_DEBUG

154 #endif
155  retval = ERR_AIN_PARSE_MISSING_KEY; /* Failed to locate a key */
156  }
157  }
158  ADC_GainCalibrate(input);
