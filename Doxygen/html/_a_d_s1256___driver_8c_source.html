585 void ADS1256_Reset_By_Clock(void) {
586  ID = 0xFFU; /* Reset the stored ID so we can properly identify the ADC after reset. */
587  ADS1256_CLK_To_GPIO(); /* Change the clock line from SPI mode to GPIO mode */
588  ADS1256_SCLK_LOW(); /* Start the clock reset sequence. */
589  Delay_us(10U); /* Start low */
590  ADS1256_SCLK_HIGH(); /* First high pulse */
591  Delay_us((uint64_t) (300U * ADS1256_CLK_PERIOD_US)); /* Timing characteristic t12, tuned because of timing errors */
592  ADS1256_SCLK_LOW(); /* First low pulse */
593  Delay_us((uint64_t) (5U * ADS1256_CLK_PERIOD_US)); /* Timing characteristic t13 */
594  ADS1256_SCLK_HIGH(); /* Second high pulse */
595  Delay_us((uint64_t) (550U * ADS1256_CLK_PERIOD_US)); /* Timing characteristic t14, tuned because of timing errors */
596  ADS1256_SCLK_LOW(); /* Second low pulse */
597  Delay_us((uint64_t) (5U * ADS1256_CLK_PERIOD_US)); /* Timing characteristic t13 */
598  ADS1256_SCLK_HIGH(); /* Third high pulse */
599  Delay_us((uint64_t) (1050U * ADS1256_CLK_PERIOD_US)); /* Timing characteristic t15, tuned because of timing errors */
600  ADS1256_SCLK_LOW(); /* End the clock reset sequence. */
601  ADS1256_GPIO_To_CLK(); /* Return the clock line to SPI mode. */
602 }

740 bool ADS1256_IsDataReady(bool useCommand) {
741  /* TODO: This method should be smart enough to determine if the DRDY pin has been enabled and default to SPI if not. */
742  if (useCommand) { /* If we are using the SPI command */
743  ADS1256_ReadRegister(ADS1256_STATUS); /* Read the status register from the device */
744  uint8_t byte = ADS1256_GetRegisterBits(ADS1256_STATUS, ADS1256_DRDY_BIT, ADS1256_DRDY_SPAN); /* Retrieve the DRDY bit from the register */
745  bool ready = false; /* Assume data is not ready */
746  if (byte == ADS1256_DATA_READY) {
747  ready = true; /* Data is ready. DRDY active low */
748  }
749 #ifdef ADS1256_DEBUG
750  printf("[ADS1256] ADS1256 Data ready: 0x%02X\n\r", byte);
751 #endif
752  return ready;
753  } else {
754  /* The intention is that false = 0 = data ready so we negate the logic */
755  return !GPIO_ReadInputDataBit(ADS1256_DRDY_GPIO_PORT, ADS1256_DRDY_PIN);
756  }
