139 void DI_Machine_Service(void) {
140  /* Determine the state */
141  switch (CurrentState) {
142  case DI_UNINITIALIZED:
143  DI_Machine_Init();
144  break;
145  case DI_INITIALIZED:
146  DI_Machine_Idle();
147  break;
148  case DI_IDLE:
149  /* We don't need to do anything, just let it run */
150  break;
151  case DI_CHANNEL_SAMPLING: {
152  Digital_Input_t* input = samplingInputs[currentSamplingInput];
153  if ((SampleCurrent < SampleTotal) || (SampleTotal == 0)) {
154  if (numberSamplingInputs == 1) {
155  SampleDigitalInput(samplingInputs[0]);
156  WriteDigitalInput(samplingInputs[0]);
157  ++SampleCurrent;
158  } else {
159  for (uint_fast8_t i = 0; i < NUM_DIGITAL_INPUTS; ++i) {
160  input = samplingInputs[i];
161  if ((input != NULL) & (input->added == CHANNEL_ADDED)) {
162  SampleDigitalInput(input);
163  WriteDigitalInput(input);
164  }
165  }
166  ++SampleCurrent;
167  }
168  } else {
169  /* Sampling has completed. */
170 #ifdef DI_STATE_MACHINE_DEBUG
171  printf("[DI STATE MACHINE] Channel sampling is complete.\n\r");
172 #endif
173  TelnetWriteStatusMessage("DI Channel sampling completed.");
174  DI_Machine_Idle();
175  CompletedDISampling();
176  }
177  }
178  break;
179  case DI_RESET:
180  SampleCurrent = 0U;
181  SampleTotal = 0U;
182  samplingInputs = NULL;
183  numberSamplingInputs = 0U;
184  DI_Machine_Idle();
185  break;
186  default:
187  /* TODO: Throw an error */
188  break;
189  }
190 }

218  /* We use explicit checking here in case other state are added later */
219  if ((CurrentState != DI_INITIALIZED) && (CurrentState != DI_CHANNEL_SAMPLING)) {

240 void DI_Machine_Input_Sample(Digital_Input_t** inputs, uint32_t count, bool singleChannel) {
241  /* We use explicit checking here in case other state are added later */
242  if ((CurrentState != DI_IDLE)) {
243 #ifdef DI_STATE_MACHINE_DEBUG
244  printf("[DI STATE MACHINE] Attempted to enter DI_CHANNEL_SAMPLING state from %s\n\r", DIMachine_StringFromState(CurrentState));
245 #endif
246  return;
247  }
248  /* Validate the inputs list */
249  if (inputs == NULL ) {
250 #ifdef DI_STATE_MACHINE_DEBUG
251  printf("[DI STATE MACHINE] Attempted to enter DI_CHANNEL_SAMPLING state with a NULL digital input list. Ignoring...\n\r");
252 #endif
253  return;
254  }
255 
256  /* Save sample count */
257  SampleCurrent = 0U;
258  SampleTotal = count;
259 
260  if (singleChannel == true) {
261  /* Validate the input(s) */
262  if (inputs[0] == NULL ) {
263 #ifdef DI_STATE_MACHINE_DEBUG
264  printf("[DI STATE MACHINE] Attempted to enter DI_CHANNEL_SAMPLING state with a NULL digital input. Ignoring...\n\r");
265 #endif
266  return;
267  }
268  if (inputs[0]->added == CHANNEL_NOTADDED) {
269 #ifdef ADC_STATE_MACHINE_DEBUG
270  printf("[DI STATE MACHINE] Attempted to enter DI_CHANNEL_SAMPLING state with an un-added digital input. Ignoring...\n\r");
271 #endif
272  return;
273  }
274  /* Select input */
275  currentSamplingInput = 0;
276  numberSamplingInputs = 1;
277  } else {
278  /* Validate the input(s) */
279  uint8_t i = 0U;
280  for (; i < NUM_DIGITAL_INPUTS; ++i) {
281  if (inputs[i] != NULL && inputs[i]->added == CHANNEL_ADDED) {
282  break;
283  }
284  }
285  if (i >= NUM_DIGITAL_INPUTS) {
286 #ifdef DI_STATE_MACHINE_DEBUG
287  printf("[DI STATE MACHINE] Attempted to enter DI_CHANNEL_SAMPLING state with NULL digital inputs. Ignoring...\n\r");
288 #endif
289  return;
290  }
291  /* Select input */
292  currentSamplingInput = i; /* Use i here so we don't loose any work from the search earlier */
293  numberSamplingInputs = NUM_DIGITAL_INPUTS;
294  }
295 
296  samplingInputs = inputs;
297  Digital_Input_t* input = samplingInputs[currentSamplingInput];
298 
299  if (input == NULL) {
300  uint8_t i = 0U;
301  while (input == NULL ) { /* Keep searching until we find a non-null input */
302  input = samplingInputs[++i];
303  if (i == NUM_DIGITAL_INPUTS) {
304  /* We have reached the end of a set */
305  /* There is nothing to sample */
306  return;
307  }
308  }
309  currentSamplingInput = i;
310  }
311  /* Save current time and sample count */
312  SampleCurrent = 0U;
313  SampleTotal = count;
314  CurrentState = DI_CHANNEL_SAMPLING;
315 }
