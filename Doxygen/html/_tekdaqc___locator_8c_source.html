66 /*
67  * An array that contains the device locator response data. The format of the
68  * data is as follows:
69  *
70  * Byte Description
71  * -------- ------------------------
72  * 0 TAG_STATUS
73  * 1 Packet Length
74  * 2 CMD_DISCOVER_TARGET
75  * 3 Board Type
76  * 4..35 Board ID (32 character serial number)
77  * 36..39 Client IP Address
78  * 40..45 MAC Address
79  * 46..49 Firmware Version
80  * 50..113 Application Title
81  * 114 Checksum
82  */
83 static unsigned char LocatorData[LOCATOR_DATA_LENGTH];
84 
85 /* The contents of the received UDP packet that will cause the Tekdaqc to return a response */
86 static unsigned char LocatorMessage[] = "TEKDAQC CONNECT";
87 
88 /* The length of the expected UDP packet message */
89 static uint8_t LengthLocatorMessage = 14U;

120 static void Tekdaqc_LocatorReceive(void *arg, struct udp_pcb *pcb, struct pbuf *p, struct ip_addr *addr, uint16_t port) {
121 #ifdef LOCATOR_DEBUG
122  printf("[Locator] Packet received.\n\r");
123 #endif
124 
125  if (TelnetIsConnected() == TRUE) {
126 #ifdef LOCATOR_DEBUG
127  printf("[Locator] Packet received while Telnet connection is active. Ignoring.\n\r");
128 #endif
129  return;
130  }
131 
132  unsigned char *data;
133  int i = 0;
134 
135  /* Validate the contents of the datagram. This expects to see a message that matches the variable LocatorMessage */
136  data = p->payload;
137  if ((p->len != LengthLocatorMessage) || (p->len < LengthLocatorMessage)) {
138  bool VALID = true;
139  for (i = 0; i < p->len; ++i) {
140  if (data[i] != LocatorMessage[i]) {
141  VALID = false;
142  break;
143  }
144  }
145  if (!VALID) {
146  pbuf_free(p);
147  return;
148  }
149  }
150 
151  /* The incoming pbuf is no longer needed, so free it. */
152  pbuf_free(p);
153 
154  /* Allocate a new pbuf for sending the response. */
155  p = pbuf_alloc(PBUF_TRANSPORT, sizeof(LocatorData), PBUF_RAM);
156  if (p == NULL ) {
157  return;
158  }
159 
160  /* Calculate and fill in the checksum on the response packet. */
161  LocatorData[sizeof(LocatorData) - 1U] = 0U;
162  for (i = 0U; i < (sizeof(LocatorData) - 1U); ++i) {
163  LocatorData[sizeof(LocatorData) - 1U] -= LocatorData[i];
164  }
165 
166  /* Copy the response packet data into the pbuf. */
167  data = p->payload;
168  for (i = 0; i < sizeof(LocatorData); ++i) {
169  data[i] = LocatorData[i];
170  }
171 
172  /* Send the response. */
173  udp_sendto(pcb, p, addr, port);
174  /* Free the pbuf. */
175  pbuf_free(p);
176 }

192 void Tekdaqc_LocatorInit(void) {
193  void *pcb;
194 
195  /* Clear out the response data. */
196  for (uint8_t i = 0U; i < LOCATOR_DATA_LENGTH; ++i) {
197  LocatorData[i] = 0U;
198  }
199 
200  /* Fill in the header for the response data. */
201  LocatorData[0] = TAG_STATUS;
202  LocatorData[1] = sizeof(LocatorData);
203  LocatorData[2] = CMD_DISCOVER_TARGET;
204 
205  /* Fill in the MAC address for the response data. */
206  LocatorData[40] = MAC_ADDR0;
207  LocatorData[41] = MAC_ADDR1;
208  LocatorData[42] = MAC_ADDR2;
209  LocatorData[43] = MAC_ADDR3;
210  LocatorData[44] = MAC_ADDR4;
211  LocatorData[45] = MAC_ADDR5;
212 
213  /* Fill in the Board Type */
214  Tekdaqc_LocatorBoardTypeSet(TEKDAQC_BOARD_TYPE );
215 
216  /* Create a new UDP port for listening to device locator requests. */
217  pcb = udp_new();
218  udp_recv(pcb, Tekdaqc_LocatorReceive, NULL);
219  udp_bind(pcb, IP_ADDR_ANY, LOCATOR_PORT);
220 }

252 void Tekdaqc_LocatorClientIPSet(uint32_t ip) {
253  /* Save the client IP address in the response data. */
254  LocatorData[36] = (unsigned char) (ip & 0xffU);

300 void LocatorAppTitleSet(const unsigned char *title) {
301  uint32_t count;
302 
303  /* Copy the application title string into the response data. */
304  for (count = 0U; (count < 64U) && *title; ++count) {
305  LocatorData[count + 19U] = *title++;
306  }
307 
308  /* Zero-fill the remainder of the space in the response data (if any). */
309  for (; count < 64U; ++count) {
310  LocatorData[count + 19U] = 0U;
311  }
312 }
