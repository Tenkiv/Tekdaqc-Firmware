116 static uint32_t ComputeOffset(ADS1256_SPS_t rate, ADS1256_PGA_t gain, ADS1256_BUFFER_t buffer, float temperature) {
117  uint32_t offset = 0;
118  switch (buffer) {
119  case ADS1256_BUFFER_ENABLED:
120  offset += CALIBRATION_BUFFER_OFFSET;
121  break;
122  case ADS1256_BUFFER_DISABLED:
123  /* No offset, do nothing */
124  break;
125  default:
126  snprintf(TOSTRING_BUFFER, sizeof(TOSTRING_BUFFER),
127  "Error computing lookup offset in calibration table for rate: %s, gain: %s, buffer: %s and temperature: %f Deg C.",
128  ADS1256_StringFromSPS(rate), ADS1256_StringFromPGA(gain), ADS1256_StringFromBuffer(buffer), temperature);
129  TelnetWriteErrorMessage(TOSTRING_BUFFER);
130  break;
131  }
132  return offset;
133 }

276 uint32_t Tekdaqc_GetGainCalibration(ADS1256_SPS_t rate, ADS1256_PGA_t gain, ADS1256_BUFFER_t buffer, float temperature) {
277  uint8_t rate_index = 0U;
278  uint8_t gain_index = 0U;
279  uint8_t buffer_index = 0U;
280  ComputeTableIndices(&rate_index, &gain_index, &buffer_index, rate, gain, buffer);
281  uint32_t baseGain = baseGainCalibrations[rate_index][gain_index][buffer_index];
282 
283  if (CALIBRATION_VALID != TRUE) {
284 #ifdef CALIBRATION_TABLE_DEBUG
285  printf("[Calibration Table] The calibration table is not valid, returning ADC calibration only (0x%" PRIX32 ").\n\r", baseGain);
286 #endif
287  return baseGain;
288  }
289  if (temperature < CAL_TEMP_LOW || temperature > CAL_TEMP_HIGH) {
290  /* The temperature is out of range, we will return the closest */
291 #ifdef CALIBRATION_TABLE_DEBUG
292  printf("[Calibration Table] The requested temperature %f was out of range. Minimum is %f and maximum is %f.\n\r", temperature,
293  CAL_TEMP_LOW, CAL_TEMP_HIGH);
294 #endif
295  snprintf(TOSTRING_BUFFER, sizeof(TOSTRING_BUFFER),
296  "Error fetching the gain calibration value for temperature: %f Deg C. Temperature out of range. Allowable range is %f to %f Deg C",
297  temperature, CAL_TEMP_LOW, CAL_TEMP_HIGH);
298  TelnetWriteErrorMessage(TOSTRING_BUFFER);
299  if (temperature < CAL_TEMP_LOW) {
300  temperature = CAL_TEMP_LOW;
301  } else {
302  temperature = CAL_TEMP_HIGH;
303  }
304  }
305 
306  uint8_t num_temp_steps = (uint8_t) (temperature / CAL_TEMP_STEP);
307  float low_temp = CAL_TEMP_LOW * num_temp_steps;
308  float high_temp = CAL_TEMP_HIGH * num_temp_steps;
309  float factor = (temperature - CAL_TEMP_LOW) / CAL_TEMP_STEP;
310 
311  uint32_t offset = ComputeOffset(rate, gain, buffer, low_temp); //Add one for the move to gain
312  uint32_t Address = CAL_DATA_START_ADDR + 4 * offset; //Multiply offset by 4 because entries are 4bytes long
313 
314  uint32_t data_low = (*(__IO uint32_t*) Address);
315 
316  offset = ComputeOffset(rate, gain, buffer, high_temp); //Add one for the move to gain
317  Address = CAL_DATA_START_ADDR + 4 * offset; //Multiply offset by 4 because entries are 4bytes long
318 
319  uint32_t data_high = (*(__IO uint32_t*) Address);
320  return (baseGain + InterpolateValue(data_low, data_high, factor));
321 }

380 FLASH_Status Tekdaqc_SetCalibrationMode(void) {
381  /* Enable the flash control register access */
382  FLASH_Unlock();
383 
384  /* Clear pending flags (if any) */
385  FLASH_ClearFlag(FLASH_FLAG_EOP | FLASH_FLAG_OPERR | FLASH_FLAG_WRPERR | FLASH_FLAG_PGAERR | FLASH_FLAG_PGPERR | FLASH_FLAG_PGSERR);
386 
387  FLASH_Status status = FLASH_COMPLETE;
388 
389  /* FLASH_OB_Unlock(); */
390 
391  /* Disable write protection for this sector */
392  /*FLASH_OB_WRPConfig(CALIBRATION_WPSECTOR, DISABLE);
393 
394  FLASH_OB_Launch(); */
395 
396  /* Erase the calibration sector */
397  status = FLASH_EraseSector(CALIBRATION_SECTOR, FLASH_VOLTAGE_RANGE);
398  if (status != FLASH_COMPLETE) {
399  return status;
400  }
401 
402  /* Program the user Flash area word by word area defined by FLASH_USER_START_ADDR and FLASH_USER_END_ADDR */
403  uint32_t Address = ADDR_CALIBRATION_BASE;
404 
405  while (Address < ADDR_CALIBRATION_END) {
406  status = FLASH_ProgramWord(Address, CALIBRATION_ERASE_DATA);
407  if (status == FLASH_COMPLETE) {
408  Address = Address + 4;
409  } else {
410  return status;
411  }
412  }
413 
414  /* Enable write protection for this sector */
415  /*FLASH_OB_WRPConfig(CALIBRATION_WPSECTOR, ENABLE);
416 
417  FLASH_OB_Launch();*/
418 
419  /* FLASH_OB_Lock(); */
420  /* Disable the flash control register access */
421 
422  CalibrationModeEnabled = true;
423  return status;
424 }

449 FLASH_Status Tekdaqc_SetSerialNumber(char* serial) {
450  FLASH_Status retval = FLASH_COMPLETE;
451  if (CalibrationModeEnabled == false) {
452  return FLASH_ERROR_WRP;
453  }
454  if (strlen(serial) < BOARD_SERIAL_NUM_LENGTH) {
455  /* The serial number isn't long enough */
456 #ifdef CALIBRATION_TABLE_DEBUG
457  printf("[Calibration Table] The provided serial number to flash is not long enough.\n\r");
458 #endif
459  retval = FLASH_ERROR_PROGRAM;
460  }
461  if (retval == FLASH_COMPLETE) {
462  for (int i = BOARD_SERIAL_NUM_ADDR; i < BOARD_SERIAL_NUM_ADDR + BOARD_SERIAL_NUM_LENGTH; i++) {
463  FLASH_Status status = FLASH_ProgramByte(i, serial[i]);
464  if (status != FLASH_COMPLETE) {
465  retval = status;
466  }
467  }
468  }
469  return retval;
470 }
