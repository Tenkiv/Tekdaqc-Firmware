58 void Timer_Config(void) {
59 #ifdef DEBUG
60  printf("[Config] Configuring SYSTICK timing.\n\r");
61 #endif
62  RCC_ClocksTypeDef RCC_Clocks;
63 
64  /* Configure Systick clock source as HCLK */
65  SysTick_CLKSourceConfig(SysTick_CLKSource_HCLK);
66 
67  /* SystTick configuration: an interrupt every 10ms */
68  RCC_GetClocksFreq(&RCC_Clocks);
69  SysTick_Config(RCC_Clocks.HCLK_Frequency / SYSTEMTICK_DIVIDER);
