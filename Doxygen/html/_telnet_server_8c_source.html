135 /*
136  * @internal
137  * This function is called when the lwIP TCP/IP stack has an incoming
138  * connection request on the telnet port.
139  *
140  * @param arg void* Argument pointer passed to the handler by the lwIP stack.
141  * @param pcb tcp_pcb* struct The PCB structure this callback is for.
142  * @param err lwIP err_t with the current error status.
143  * @retval err lwIP err_t with the result of the this function.
144  */
145 static err_t TelnetAccept(void *arg, struct tcp_pcb *pcb, err_t err) {
146 #ifdef TELNET_DEBUG
147  printf("[Telnet Server] Incoming connection requested on Telnet port.\n\r");
148 #endif
149  err_t ret_err;
150 
151  LWIP_UNUSED_ARG(arg);
152  LWIP_UNUSED_ARG(err);
153 
154  /* Check if already connected. */
155  if (TelnetIsConnected() == true) {
156  /* There is already a connected client, so refuse this connection with
157  a message indicating this fact. */
158 #ifdef TELNET_DEBUG
159  printf("[Telnet Server] A connection was attempted while an active connection is open.\n\r");
160 #endif
161  tcp_accepted(pcb);
162  tcp_arg(pcb, NULL);
163  tcp_sent(pcb, TelnetSent);
164  tcp_write(pcb, ErrorMessage, sizeof(ErrorMessage), 1);
165  tcp_output(pcb);
166  /* Temporarily accept this connection until the message is transmitted. */
167  return (ERR_OK);
168  }
169 
170  /* Setup the TCP connection priority. */
171  tcp_setprio(pcb, TCP_PRIO_MIN);
172 
173  CreateTelnetServer();
174 #ifdef TELNET_DEBUG
175  printf("[Telnet Server] Initializing telnet server.\n\r");
176 #endif
177  telnet_server.pcb = pcb;
178  telnet_server.pcb->so_options |= SOF_KEEPALIVE;
179  telnet_server.pcb->keep_idle = 5000;
180  telnet_server.pcb->keep_intvl = 1000;
181  telnet_server.pcb->keep_cnt = 5;
182 
183  /* Mark that a client has connected. */
184  IsConnected = true;
185  /* Accept this connection. */
186  tcp_accepted(pcb);
187 #ifdef TELNET_DEBUG
188  printf("[Telnet Server] An incoming connection was accepted.\n\r");
189 #endif
190 
191  /* Setup the TCP callback argument. */
192  tcp_arg(pcb, &telnet_server);
193 
194  /* Initialize lwIP tcp_recv callback function for pcb */
195  tcp_recv(pcb, TelnetReceive);
196 
197  /* Initialize lwIP tcp_err callback function for pcb */
198  tcp_err(pcb, TelnetError);
199 
200  /* Initialize lwIP tcp_poll callback function for pcb */
201  tcp_poll(pcb, TelnetPoll, 1);
202 
203  /* Setup the TCP sent callback function. */
204  tcp_sent(pcb, TelnetSent);
205  /* Initialize the count of outstanding bytes. The initial byte acked as
206  part of the SYN -> SYN/ACK sequence is included so that the byte count
207  works out correctly at the end. */
208  telnet_server.outstanding = sizeof(TelnetInit) + 1;
209  /* Do not close the telnet connection until requested. */
210  telnet_server.close = 0;
211 #ifdef TELNET_DEBUG
212  printf("[Telnet Server] Writing the init messages\n\r");
213 #endif
214  /* Send the telnet initialization string. */
215  tcp_write(pcb, TelnetInit, sizeof(TelnetInit), 1);
216  tcp_output(pcb);
217 
218  TelnetWriteDebugMessage("[TELNET] Telnet Server Connected. Welcome.");
219 
220  /* Return a success code. */
221  ret_err = ERR_OK;
222  return ret_err;
223 }

225 /*
226  * @internal
227  * This function is called when the lwIP TCP/IP stack has an incoming packet to
228  * be processed.
229  *
230  * @param arg void* Argument pointer passed to the handler by the lwIP stack.
231  * @param pcb tcp_pcb* struct The PCB structure this callback is for.
232  * @param p pbuf* struct The data buffer from the lwIP stack.
233  * @param err lwIP err_t with the current error status.
234  * @retval err lwIP err_t with the result of the this function.
235  */
236 static err_t TelnetReceive(void *arg, struct tcp_pcb *pcb, struct pbuf *p,
237  err_t err) {
238  struct pbuf *q;
239  unsigned long ulIdx;
240  unsigned char *pucData;
241  TelnetServer_t* server;
242  if (arg != NULL) {
243  server = (TelnetServer_t*) arg;
244 
245  /* Process the incoming packet. */
246  if ((err == ERR_OK) && (p != NULL)) {
247 #ifdef TELNET_DEBUG
248  printf("[Telnet Server] Processing received packet.\n\r");
249 #endif
250  /* Accept the packet from TCP. */
251  tcp_recved(pcb, p->tot_len);
252  /* Loop through the pbufs in this packet. */
253  for (q = p, pucData = (unsigned char*) q->payload; q != NULL;
254  q = q->next) {
255  /* Loop through the bytes in this pbuf. */
256  for (ulIdx = 0; ulIdx < q->len; ulIdx++) {
257  /* Process this character. */
258  TelnetProcessCharacter(pucData[ulIdx]);
259  }
260  }
261  /* Free the pbuf. */
262  pbuf_free(p);
263  } else if ((err == ERR_OK) && (p == NULL)) {
264  /* If a null packet is passed in, close the connection. */
265  server->length = 0;
266  TelnetClose();
267  }
268  } else {
269 #ifdef TELNET_DEBUG
270  printf("[Telnet Server] Could not cast receive args to telnet server struct because they were null.\n\r");
271  return ERR_VAL;
272 #endif
273  }
274  /* Return okay. */
275  return (ERR_OK);
276 }

288 static err_t TelnetSent(void *arg, struct tcp_pcb *pcb, u16_t len) {
289  /* See if this is for the game connection or for a secondary connection. */
290  if (arg) {
291  TelnetServer_t* server = (TelnetServer_t*) arg;
292 #ifdef TELNET_DEBUG
293  printf("[Telnet Server] Sent packet was acknowledged.\n\r");
294 #endif
295  /* Decrement the count of outstanding bytes. */
296  server->outstanding -= len;
297  } else {
298  /* See if this is the ACK for the error message. */
299  if (len == sizeof(ErrorMessage)) {
300  /* Close this telnet connection now that the error message has been transmitted. */
301  tcp_sent(pcb, 0);
302  tcp_close(pcb);
303  }
304  }
305  /* Return OK. */
306  return (ERR_OK);
307 }

387 
388  if (telnet_pcb != NULL) {
389 #ifdef TELNET_DEBUG

391 #endif

394  err = tcp_bind(telnet_pcb, IP_ADDR_ANY, TELNET_PORT);
395 
396  if (err == ERR_OK) {

408 #ifdef TELNET_DEBUG

410 #endif

412  }
413  } else {

469 err_t TelnetPoll(void *arg, struct tcp_pcb *tpcb) {
470  err_t ret_err;
471  TelnetServer_t* server;
472  server = (TelnetServer_t*) arg;
473 
474  if (server != NULL) {
475  unsigned long length = server->length;
476  if ((server->pcb != NULL) && (length != 0)) {
477  /* Write the data from the transmit buffer. */
478  tcp_write(server->pcb, server->buffer, length, 1);
479 
480  /* Increment the count of outstanding bytes. */
481  server->outstanding += length;
482 
483  /* Output the telnet data. */
484  tcp_output(server->pcb);
485 
486  /* Reset the size of the data in the transmit buffer. */
487  server->length = 0;
488  }
489  /* See if the telnet connection should be closed; this will only occur once
490  all transmitted data has been ACKed by the client (so that some or all
491  of the final message is not lost). */
492  if (server->pcb && (server->outstanding == 0) && (server->close != 0)) {
493 #ifdef TELNET_DEBUG
494  printf("[Telnet Server] Telnet server should be closed.\n\r");
495 #endif
496  TelnetClose();
497  }
498  ret_err = ERR_OK;
499  } else {
500 #ifdef TELNET_DEBUG
501  printf("[Telnet Server] Cannot process poll request due to null server structure.\n\r");
502 #endif
503  /* Nothing to be done */
504  tcp_abort(tpcb);
505  ret_err = ERR_ABRT;
506  }
507  return ret_err;
508 }

516 void TelnetRecvBufferWrite(char character) {
517  unsigned long ulWrite;
518  /* Ignore this character if it is the NULL character. */
519  if (character == 0) {
520 #ifdef TELNET_CHAR_DEBUG
521  printf("[Telnet Server] Ignorning NULL character.\n\r");
522 #endif
523  return;
524  }
525 
526  /* Ignore this character if it is the second part of a CR/LF or LF/CR sequence. */
527  if (((character == '\r') && (telnet_server.previous == '\n'))
528  || ((character == '\n') && (telnet_server.previous == '\r'))) {
529  return;
530  }
531 
532  /* Store this character into the receive buffer if there is space for it. */
533  ulWrite = telnet_server.recvWrite;
534  if (((ulWrite + 1) % sizeof(telnet_server.recvBuffer))
535  != telnet_server.recvRead) {
536  telnet_server.recvBuffer[ulWrite] = character;
537  telnet_server.recvWrite = (ulWrite + 1)
538  % sizeof(telnet_server.recvBuffer);
539 #ifdef TELNET_CHAR_DEBUG
540  for (int i = 0; i <= telnet_server.recvWrite; ++i) {
541  printf("%c", telnet_server.recvBuffer[i]);
542  }
543  printf("\n\r");
544 #endif
545  } else {
546 #ifdef TELNET_DEBUG
547  printf("[Telnet Server] Could not store new character because buffer was full.\n\r");
548 #endif
549  }
550 
551  /* Save this character as the previously received telnet character. */
552  telnet_server.previous = character;
553 }

561 char TelnetRead(void) {
562  uint32_t read;
563  char ret;
564  /* Return a NULL if there is no data in the receive buffer. */
565  read = telnet_server.recvRead;
566  if (read == telnet_server.recvWrite) {
567  return (0);
568  }
569  /* Read the next byte from the receive buffer. */
570  ret = telnet_server.recvBuffer[read];
571  telnet_server.recvRead = (read + 1) % sizeof(telnet_server.recvBuffer);
572  /* Return the byte that was read. */
573  return (ret);
574 }

626 void TelnetProcessWill(char option) {
627  unsigned long ulIdx;
628 #ifdef TELNET_CHAR_DEBUG
629  printf("[Telnet Server] Processing WILL command with option: %c/0x%02X\n\r", option, option);
630 #endif
631  /* Loop through the known options. */
632  for (ulIdx = 0; ulIdx < (sizeof(TelnetOptions) / sizeof(TelnetOptions[0]));
633  ulIdx++) {
634  /* See if this option matches the option in question. */
635  if (TelnetOptions[ulIdx].option == option) {
636  /* See if the WILL flag for this option has already been set. */
637  if (((TelnetOptions[ulIdx].flags >> OPT_FLAG_WILL) & 0x01) == 0) {
638  /* Set the WILL flag for this option. */
639  TelnetOptions[ulIdx].flags = (TelnetOptions[ulIdx].flags & 0xFD)
640  | (0x01 << OPT_FLAG_WILL);
641  /* Send a DO response to this option. */
642  telnet_server.buffer[telnet_server.length++] = TELNET_IAC;
643  telnet_server.buffer[telnet_server.length++] = TELNET_DO;
644  telnet_server.buffer[telnet_server.length++] = option;
645  }
646  /* Return without any further processing. */
647  return;
648  }
649  }
650 
651  /* This option is not recognized, so send a DONT response. */
652  telnet_server.buffer[telnet_server.length++] = TELNET_IAC;
653  telnet_server.buffer[telnet_server.length++] = TELNET_DONT;
654  telnet_server.buffer[telnet_server.length++] = option;
655 }

784 /*
785  * This function processes a character received from the telnet port, handling
786  * the interpretation of telnet commands (as indicated by the telnet interpret
787  * as command (IAC) byte).
788  *
789  * @param character char The character to process.
790  * @retval none
791  */
792 void TelnetProcessCharacter(char character) {
793 #ifdef TELNET_CHAR_DEBUG
794  printf("[Telnet Server] Processing Character: %c/0x%02X\n\r", character, character);
795 #endif
796  /* Determine the current state of the telnet command parser. */
797  switch (telnet_server.state) {
798  /* The normal state of the parser, were each character is either sent
799  to the UART or is a telnet IAC character. */
800  case STATE_NORMAL: {
801  /* See if this character is the IAC character. */
802  if (character == TELNET_IAC) {
803  /* Skip this character and go to the IAC state. */
804  telnet_server.state = STATE_IAC;
805  } else {
806  /* Write this character to the receive buffer. */
807  TelnetRecvBufferWrite(character);
808  /* Echo this character */
809  TelnetWrite(character); //Echo the character back
810  }
811  break;
812  }
813  /* The previous character was the IAC character. */
