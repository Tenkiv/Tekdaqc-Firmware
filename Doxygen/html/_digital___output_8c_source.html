150 static void SetDigitalOutputState(Digital_Output_t* output, DigitalLevel_t level) {
151  /* Set the state */
152  uint8_t control = TLE7232_ReadRegister(TLE7232_REG_CTL, output->output % 8U);
153  uint8_t channel = output->output;
154  if (channel >= 8) { /* Convert the channel to a single chip index */
155  channel -= 8;
156  }
157  uint8_t mask = 0x01 << channel; /* Move the bit flag to the correct position */
158  if (level == OUTPUT_ON) {
159  control |= mask; /* Set the appropriate bit */
160  } else {
161  control ^= mask; /* Clear the appropriate bit */
162  }
163  TLE7232_WriteRegister(TLE7232_REG_CTL, control, output->output % 8U); /* Update the register */
164 }

237 Tekdaqc_Function_Error_t CreateDigitalOutput(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], int count) {
238  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
239  char* param;
240  char* testPtr = NULL;
241  int8_t index = -1;
242  uint8_t output = NULL_CHANNEL; /* The physical input */
243  uint8_t out = NULL_CHANNEL;
244  char name[MAX_DIGITAL_OUTPUT_NAME_LENGTH]; /* The name */
245  strcpy(name, "NONE");
246  for (int i = 0; i < NUM_ADD_DIGITAL_OUTPUT_PARAMS; ++i) {
247  index = GetIndexOfArgument(keys, ADD_DIGITAL_OUTPUT_PARAMS[i], count);
248  if (index >= 0) { /* We found the key in the list */
249  param = values[index]; /* We use the discovered index for this key */
250  switch (i) { /* Switch on the key not position in arguments list */
251  case 0U: /* OUTPUT key */
252  out = (uint8_t) strtol(param, &testPtr, 10);
253  if (testPtr == param) {
254  retval = ERR_DOUT_PARSE_ERROR;
255  } else {
256  if (out >= 0U && out <= NUM_DIGITAL_OUTPUTS) {
257  /* A valid output number */
258  output = out;
259  } else {
260  /* Input number out of range */
261 #ifdef DIGITALOUTPUT_DEBUG
262  printf("[Digital Output] The requested output number is invalid.\n\r");
263 #endif
264  retval = ERR_DOUT_OUTPUT_OUTOFRANGE;
265  }
266  }
267  break;
268  case 1U: /* NAME key */
269  strcpy(name, param);
270  break;
271  default:
272  /* Return an error */
273  retval = ERR_DOUT_PARSE_ERROR;
274  }
275  } else if (i == 1U) {
276  /* The NAME key is not strictly required, apply the defaults */
277  continue;
278  } else {
279  /* Somehow an error happened */
280 #ifdef DIGITALOUTPUT_DEBUG
281  printf("[Digital Output] Unable to locate required key: %s\n\r", ADD_DIGITAL_OUTPUT_PARAMS[i]);
282 #endif
283  retval = ERR_DOUT_PARSE_MISSING_KEY; /* Failed to locate a key */
284  }
285  if (retval != ERR_FUNCTION_OK) {
286  break;
287  }
288  }
289  if (retval == ERR_FUNCTION_OK) {
290  if (output != NULL_CHANNEL) {
291  Digital_Output_t* dig_output = GetDigitalOutputByNumber(output);
292  if (dig_output != NULL) {
293  if (dig_output->added != CHANNEL_ADDED) {
294  dig_output->output = output;
295  strcpy(dig_output->name, name);
296  dig_output->level = LOGIC_LOW;
297  dig_output->timestamp = 0U;
298  retval = AddDigitalOutput(dig_output);
299  } else {
300  retval = ERR_DOUT_OUTPUT_EXISTS;
301  }
302  } else {
303  /* The output could not be found */
304  retval = ERR_DOUT_OUTPUT_NOT_FOUND;
305  }
306  } else {
307  /* A valid input was never specified */
308  retval = ERR_DOUT_OUTPUT_UNSPECIFIED;
309  }
310  }
311  return retval;
312 }

406 Tekdaqc_Function_Error_t SetDigitalOutput(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
407  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
408  char* param;
409  int8_t index = -1;
410  uint8_t output = NULL_CHANNEL; /* The physical input */
411  DigitalLevel_t level = OUTPUT_OFF;
412  for (int i = 0; i < NUM_SET_DIGITAL_OUTPUT_PARAMS; ++i) {
413  index = GetIndexOfArgument(keys, SET_DIGITAL_OUTPUT_PARAMS[i], count);
414  if (index >= 0) { /* We found the key in the list */
415  param = values[index]; /* We use the discovered index for this key */
416  switch (i) { /* Switch on the key not position in arguments list */
417  case 0: { /* OUTPUT key */
418  char* testPtr = NULL;
419  uint8_t out = (uint8_t) strtol(param, &testPtr, 10);
420  if (testPtr == param) {
421  retval = ERR_DOUT_PARSE_ERROR;
422  } else {
423  if (out >= 0U && out <= NUM_DIGITAL_OUTPUTS) {
424  /* A valid input number */
425  output = out;
426  } else {
427  /* Input number out of range */
428 #ifdef DIGITALOUTPUT_DEBUG
429  printf("[Digital Output] The requested output number is invalid.\n\r");
430 #endif
431  retval = ERR_DOUT_OUTPUT_OUTOFRANGE;
432  }
433  }
434  break;
435  }
436  case 1: /* STATE key */
437  if (strcmp(param, "ON") == 0) {
438  level = OUTPUT_ON;
439  } else {
440  level = OUTPUT_OFF;
441  }
442  break;
443  default:
444  /* Return an error */
445  retval = ERR_DOUT_PARSE_ERROR;
446  }
447  } else {
448 #ifdef DIGITALOUTPUT_DEBUG
449  printf("[Digital Output] Unable to locate required key: %s\n\r", SET_DIGITAL_OUTPUT_PARAMS[i]);
450 #endif
451  retval = ERR_DOUT_PARSE_MISSING_KEY; /* Failed to locate a key */
452  }
453  if (retval != ERR_FUNCTION_OK) {
454  break;
455  }
456  }
457  if (retval == ERR_FUNCTION_OK) {
458  if (output != NULL_CHANNEL) {
459  Digital_Output_t* dig_output = GetDigitalOutputByNumber(output);
460  if (dig_output != NULL) {
461  if (dig_output->added == CHANNEL_ADDED) {
462  SetDigitalOutputState(dig_output, level);
463  } else {
464 #ifdef DIGITALOUTPUT_DEBUG
465  printf("[Digital Output] Tried to change the state of an output which has not been added.\n\r");
466 #endif
467  retval = ERR_DOUT_DOES_NOT_EXIST;

470  /* The output could not be found */

472  }
473  } else {
474  /* A valid input was never specified */
475  retval = ERR_DOUT_OUTPUT_UNSPECIFIED;
476  }
477  }
478  return retval;
