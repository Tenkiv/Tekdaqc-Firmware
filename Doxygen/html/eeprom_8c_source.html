52 uint16_t EE_Init(void) {
53 
54  /* Unlock the Flash Program Erase controller */
55  FLASH_Unlock();
56 
57  /* We use a uint16_t counter here because that is the width of the address field, thus that many are supported. */
58  for (uint16_t i = 0; i < NUM_EEPROM_ADDRESSES; ++i) {
59  EEPROM_ADDRESSES[i] = i; /* We will always do them sequentially */
60  }
61 
62  uint16_t PageStatus0 = 6, PageStatus1 = 6;
63  uint16_t VarIdx = 0;
64  uint16_t EepromStatus = 0, ReadStatus = 0;
65  int16_t x = -1;
66  uint16_t FlashStatus;
67 
68  /* Get Page0 status */
69  PageStatus0 = (*(__IO uint16_t*) PAGE0_BASE_ADDRESS );
70  /* Get Page1 status */
71  PageStatus1 = (*(__IO uint16_t*) PAGE1_BASE_ADDRESS );
72 
73  /* Check for invalid header states and repair if necessary */
74  switch (PageStatus0) {
75  case ERASED :
76  if (PageStatus1 == VALID_PAGE ) /* Page0 erased, Page1 valid */
77  {
78  /* Erase Page0 */
79  FlashStatus = FLASH_EraseSector(PAGE0_ID, FLASH_VOLTAGE_RANGE );
80  /* If erase operation was failed, a Flash error code is returned */
81  if (FlashStatus != FLASH_COMPLETE) {
82  return FlashStatus;
83  }
84  } else if (PageStatus1 == RECEIVE_DATA ) /* Page0 erased, Page1 receive */
85  {
86  /* Erase Page0 */
87  FlashStatus = FLASH_EraseSector(PAGE0_ID, FLASH_VOLTAGE_RANGE );
88  /* If erase operation was failed, a Flash error code is returned */
89  if (FlashStatus != FLASH_COMPLETE) {
90  return FlashStatus;
91  }
92  /* Mark Page1 as valid */
93  FlashStatus = FLASH_ProgramHalfWord(PAGE1_BASE_ADDRESS, VALID_PAGE );
94  /* If program operation was failed, a Flash error code is returned */
95  if (FlashStatus != FLASH_COMPLETE) {
96  return FlashStatus;
97  }
98  } else /* First EEPROM access (Page0&1 are erased) or invalid state -> format EEPROM */
99  {
100  /* Erase both Page0 and Page1 and set Page0 as valid page */
101  FlashStatus = EE_Format();
102  /* If erase/program operation was failed, a Flash error code is returned */
103  if (FlashStatus != FLASH_COMPLETE) {
104  return FlashStatus;
105  }
106  }
107  break;
108 
109  case RECEIVE_DATA :
110  if (PageStatus1 == VALID_PAGE ) /* Page0 receive, Page1 valid */
111  {
112  /* Transfer data from Page1 to Page0 */
113  for (VarIdx = 0; VarIdx < NUM_EEPROM_ADDRESSES ; VarIdx++) {
114  if ((*(__IO uint16_t*) (PAGE0_BASE_ADDRESS + 6)) == EEPROM_ADDRESSES[VarIdx]) {
115  x = VarIdx;
116  }
117  if (VarIdx != x) {
118  /* Read the last variables' updates */
119  ReadStatus = EE_ReadVariable(EEPROM_ADDRESSES[VarIdx], &DataVar);
120  /* In case variable corresponding to the virtual address was found */
121  if (ReadStatus != 0x1) {
122  /* Transfer the variable to the Page0 */
123  EepromStatus = EE_VerifyPageFullWriteVariable(EEPROM_ADDRESSES[VarIdx], DataVar);
124  /* If program operation was failed, a Flash error code is returned */
125  if (EepromStatus != FLASH_COMPLETE) {
126  return EepromStatus;
127  }
128  }
129  }
130  }
131  /* Mark Page0 as valid */
132  FlashStatus = FLASH_ProgramHalfWord(PAGE0_BASE_ADDRESS, VALID_PAGE );
133  /* If program operation was failed, a Flash error code is returned */
134  if (FlashStatus != FLASH_COMPLETE) {
135  return FlashStatus;
136  }
137  /* Erase Page1 */
138  FlashStatus = FLASH_EraseSector(PAGE1_ID, FLASH_VOLTAGE_RANGE );
139  /* If erase operation was failed, a Flash error code is returned */
140  if (FlashStatus != FLASH_COMPLETE) {
141  return FlashStatus;
142  }
143  } else if (PageStatus1 == ERASED ) /* Page0 receive, Page1 erased */
144  {
145  /* Erase Page1 */
146  FlashStatus = FLASH_EraseSector(PAGE1_ID, FLASH_VOLTAGE_RANGE );
147  /* If erase operation was failed, a Flash error code is returned */
148  if (FlashStatus != FLASH_COMPLETE) {
149  return FlashStatus;
150  }
151  /* Mark Page0 as valid */
152  FlashStatus = FLASH_ProgramHalfWord(PAGE0_BASE_ADDRESS, VALID_PAGE );
153  /* If program operation was failed, a Flash error code is returned */
154  if (FlashStatus != FLASH_COMPLETE) {
155  return FlashStatus;
156  }
157  } else /* Invalid state -> format eeprom */
158  {
159  /* Erase both Page0 and Page1 and set Page0 as valid page */
160  FlashStatus = EE_Format();
161  /* If erase/program operation was failed, a Flash error code is returned */
162  if (FlashStatus != FLASH_COMPLETE) {
163  return FlashStatus;
164  }
165  }
166  break;
167 
168  case VALID_PAGE :
169  if (PageStatus1 == VALID_PAGE ) /* Invalid state -> format eeprom */
170  {
171  /* Erase both Page0 and Page1 and set Page0 as valid page */
172  FlashStatus = EE_Format();
173  /* If erase/program operation was failed, a Flash error code is returned */
174  if (FlashStatus != FLASH_COMPLETE) {
175  return FlashStatus;
176  }
177  } else if (PageStatus1 == ERASED ) /* Page0 valid, Page1 erased */
178  {
179  /* Erase Page1 */
180  FlashStatus = FLASH_EraseSector(PAGE1_ID, FLASH_VOLTAGE_RANGE );
181  /* If erase operation was failed, a Flash error code is returned */
182  if (FlashStatus != FLASH_COMPLETE) {
183  return FlashStatus;
184  }
185  } else /* Page0 valid, Page1 receive */
186  {
187  /* Transfer data from Page0 to Page1 */
188  for (VarIdx = 0; VarIdx < NUM_EEPROM_ADDRESSES ; VarIdx++) {
189  if ((*(__IO uint16_t*) (PAGE1_BASE_ADDRESS + 6)) == EEPROM_ADDRESSES[VarIdx]) {
190  x = VarIdx;
191  }
192  if (VarIdx != x) {
193  /* Read the last variables' updates */
194  ReadStatus = EE_ReadVariable(EEPROM_ADDRESSES[VarIdx], &DataVar);
195  /* In case variable corresponding to the virtual address was found */
196  if (ReadStatus != 0x1) {
197  /* Transfer the variable to the Page1 */
198  EepromStatus = EE_VerifyPageFullWriteVariable(EEPROM_ADDRESSES[VarIdx], DataVar);
199  /* If program operation was failed, a Flash error code is returned */
200  if (EepromStatus != FLASH_COMPLETE) {
201  return EepromStatus;
202  }
203  }
204  }
205  }
206  /* Mark Page1 as valid */
207  FlashStatus = FLASH_ProgramHalfWord(PAGE1_BASE_ADDRESS, VALID_PAGE );
208  /* If program operation was failed, a Flash error code is returned */
209  if (FlashStatus != FLASH_COMPLETE) {
210  return FlashStatus;
211  }
212  /* Erase Page0 */
213  FlashStatus = FLASH_EraseSector(PAGE0_ID, FLASH_VOLTAGE_RANGE );
214  /* If erase operation was failed, a Flash error code is returned */
215  if (FlashStatus != FLASH_COMPLETE) {
216  return FlashStatus;
217  }
218  }
219  break;
220 
221  default: /* Any other state -> format eeprom */
222  /* Erase both Page0 and Page1 and set Page0 as valid page */
223  FlashStatus = EE_Format();
224  /* If erase/program operation was failed, a Flash error code is returned */
225  if (FlashStatus != FLASH_COMPLETE) {
226  return FlashStatus;
227  }
228  break;
229  }

247 uint16_t EE_ReadVariable(uint16_t VirtAddress, uint16_t* Data) {
248  uint16_t ValidPage = PAGE0;
249  uint16_t AddressValue = 0x5555, ReadStatus = 1;
250  uint32_t Address = EEPROM_START_ADDRESS, PageStartAddress = EEPROM_START_ADDRESS;
251 
252  /* Get active Page for read operation */
253  ValidPage = EE_FindValidPage(READ_FROM_VALID_PAGE );
254 
255  /* Check if there is no valid page */
256  if (ValidPage == NO_VALID_PAGE ) {
257  return NO_VALID_PAGE ;
258  }
259 
260  /* Get the valid Page start Address */
261  PageStartAddress = (uint32_t) (EEPROM_START_ADDRESS + (uint32_t) (ValidPage * PAGE_SIZE ));
262 
263  /* Get the valid Page end Address */
264  Address = (uint32_t) ((EEPROM_START_ADDRESS - 2) + (uint32_t) ((1 + ValidPage) * PAGE_SIZE ));
265 
266  /* Check each active page address starting from end */
267  while (Address > (PageStartAddress + 2)) {
268  /* Get the current location content to be compared with virtual address */
269  AddressValue = (*(__IO uint16_t*) Address);
270 
271  /* Compare the read address with the virtual address */
272  if (AddressValue == VirtAddress) {
273  /* Get content of Address-2 which is variable value */
274  *Data = (*(__IO uint16_t*) (Address - 2));
275 
276  /* In case variable value is read, reset ReadStatus flag */
277  ReadStatus = 0;
278 
279  break;
280  } else {
281  /* Next address location */
282  Address = Address - 4;
283  }
284  }
285 
286  /* Return ReadStatus value: (0: variable exist, 1: variable doesn't exist) */
287  return ReadStatus;
288 }

300 uint16_t EE_WriteVariable(uint16_t VirtAddress, uint16_t Data) {
301  uint16_t Status = 0;
302 
303  /* Write the variable virtual address and value in the EEPROM */
304  Status = EE_VerifyPageFullWriteVariable(VirtAddress, Data);
305 
306  /* In case the EEPROM active page is full */
307  if (Status == PAGE_FULL ) {
308  /* Perform Page transfer */

310  }
311 
312  /* Return last operation status */

357 static uint16_t EE_FindValidPage(uint8_t Operation) {
358  uint16_t PageStatus0 = 6, PageStatus1 = 6;
359 
360  /* Get Page0 actual status */
361  PageStatus0 = (*(__IO uint16_t*) PAGE0_BASE_ADDRESS );
362 
363  /* Get Page1 actual status */
364  PageStatus1 = (*(__IO uint16_t*) PAGE1_BASE_ADDRESS );
365 
366  /* Write or read operation */
367  switch (Operation) {
368  case WRITE_IN_VALID_PAGE : /* ---- Write operation ---- */
369  if (PageStatus1 == VALID_PAGE ) {
370  /* Page0 receiving data */
371  if (PageStatus0 == RECEIVE_DATA ) {
372  return PAGE0 ; /* Page0 valid */
373  } else {
374  return PAGE1 ; /* Page1 valid */
375  }
376  } else if (PageStatus0 == VALID_PAGE ) {
377  /* Page1 receiving data */
378  if (PageStatus1 == RECEIVE_DATA ) {
379  return PAGE1 ; /* Page1 valid */
380  } else {
381  return PAGE0 ; /* Page0 valid */
382  }
383  } else {
384  return NO_VALID_PAGE ; /* No valid Page */
385  }
386 
387  case READ_FROM_VALID_PAGE : /* ---- Read operation ---- */
388  if (PageStatus0 == VALID_PAGE ) {
389  return PAGE0 ; /* Page0 valid */
390  } else if (PageStatus1 == VALID_PAGE ) {
391  return PAGE1 ; /* Page1 valid */
392  } else {
393  return NO_VALID_PAGE ; /* No valid Page */
394  }
395 
396  default:
397  return PAGE0 ; /* Page0 valid */
398  }
399 }

411 static uint16_t EE_VerifyPageFullWriteVariable(uint16_t VirtAddress, uint16_t Data) {
412  FLASH_Status FlashStatus = FLASH_COMPLETE;
413  uint16_t ValidPage = PAGE0;
414  uint32_t Address = EEPROM_START_ADDRESS, PageEndAddress = EEPROM_START_ADDRESS + PAGE_SIZE;
415 
416  /* Get valid Page for write operation */
417  ValidPage = EE_FindValidPage(WRITE_IN_VALID_PAGE );
418 
419  /* Check if there is no valid page */
420  if (ValidPage == NO_VALID_PAGE ) {
421  return NO_VALID_PAGE ;
422  }
423 
424  /* Get the valid Page start Address */
425  Address = (uint32_t) (EEPROM_START_ADDRESS + (uint32_t) (ValidPage * PAGE_SIZE ));
426 
427  /* Get the valid Page end Address */
428  PageEndAddress = (uint32_t) ((EEPROM_START_ADDRESS - 2) + (uint32_t) ((1 + ValidPage) * PAGE_SIZE ));
429 
430  /* Check each active page address starting from begining */
431  while (Address < PageEndAddress) {
432  /* Verify if Address and Address+2 contents are 0xFFFFFFFF */
433  if ((*(__IO uint32_t*) Address) == 0xFFFFFFFF) {
434  /* Set variable data */
435  FlashStatus = FLASH_ProgramHalfWord(Address, Data);
436  /* If program operation was failed, a Flash error code is returned */
437  if (FlashStatus != FLASH_COMPLETE) {
438  return FlashStatus;
439  }
440  /* Set variable virtual address */
441  FlashStatus = FLASH_ProgramHalfWord(Address + 2, VirtAddress);
442  /* Return program operation status */
443  return FlashStatus;
444  } else {
445  /* Next address location */
446  Address = Address + 4;
447  }
448  }
449 
450  /* Return PAGE_FULL in case the valid page is full */
451  return PAGE_FULL ;
452 }

465 static uint16_t EE_PageTransfer(uint16_t VirtAddress, uint16_t Data) {
466  FLASH_Status FlashStatus = FLASH_COMPLETE;
467  uint32_t NewPageAddress = EEPROM_START_ADDRESS;
468  uint16_t OldPageId = 0;
469  uint16_t ValidPage = PAGE0, VarIdx = 0;
470  uint16_t EepromStatus = 0, ReadStatus = 0;
471 
472  /* Get active Page for read operation */
473  ValidPage = EE_FindValidPage(READ_FROM_VALID_PAGE );
474 
475  if (ValidPage == PAGE1 ) /* Page1 valid */
476  {
477  /* New page address where variable will be moved to */
478  NewPageAddress = PAGE0_BASE_ADDRESS;
479 
480  /* Old page ID where variable will be taken from */
481  OldPageId = PAGE1_ID;
482  } else if (ValidPage == PAGE0 ) /* Page0 valid */
483  {
484  /* New page address where variable will be moved to */
485  NewPageAddress = PAGE1_BASE_ADDRESS;
486 
487  /* Old page ID where variable will be taken from */
488  OldPageId = PAGE0_ID;
489  } else {
490  return NO_VALID_PAGE ; /* No valid Page */
491  }
492 
493  /* Set the new Page status to RECEIVE_DATA status */
494  FlashStatus = FLASH_ProgramHalfWord(NewPageAddress, RECEIVE_DATA );
495  /* If program operation was failed, a Flash error code is returned */
496  if (FlashStatus != FLASH_COMPLETE) {
497  return FlashStatus;
498  }
499 
500  /* Write the variable passed as parameter in the new active page */
501  EepromStatus = EE_VerifyPageFullWriteVariable(VirtAddress, Data);
502  /* If program operation was failed, a Flash error code is returned */
503  if (EepromStatus != FLASH_COMPLETE) {
504  return EepromStatus;
505  }
506 
507  /* Transfer process: transfer variables from old to the new active page */
508  for (VarIdx = 0; VarIdx < NUM_EEPROM_ADDRESSES ; VarIdx++) {
509  if (EEPROM_ADDRESSES[VarIdx] != VirtAddress) /* Check each variable except the one passed as parameter */
510  {
511  /* Read the other last variable updates */
512  ReadStatus = EE_ReadVariable(EEPROM_ADDRESSES[VarIdx], &DataVar);
513  /* In case variable corresponding to the virtual address was found */
514  if (ReadStatus != 0x1) {
515  /* Transfer the variable to the new active page */
516  EepromStatus = EE_VerifyPageFullWriteVariable(EEPROM_ADDRESSES[VarIdx], DataVar);
517  /* If program operation was failed, a Flash error code is returned */
518  if (EepromStatus != FLASH_COMPLETE) {
519  return EepromStatus;
520  }
521  }
522  }
523  }
524 
525  /* Erase the old Page: Set old Page status to ERASED status */
526  FlashStatus = FLASH_EraseSector(OldPageId, FLASH_VOLTAGE_RANGE );
527  /* If erase operation was failed, a Flash error code is returned */
528  if (FlashStatus != FLASH_COMPLETE) {
529  return FlashStatus;
530  }
531 
532  /* Set new Page status to VALID_PAGE status */
533  FlashStatus = FLASH_ProgramHalfWord(NewPageAddress, VALID_PAGE );
534  /* If program operation was failed, a Flash error code is returned */
535  if (FlashStatus != FLASH_COMPLETE) {
536  return FlashStatus;
537  }
538 
539  /* Return last operation flash status */
540  return FlashStatus;
541 }
