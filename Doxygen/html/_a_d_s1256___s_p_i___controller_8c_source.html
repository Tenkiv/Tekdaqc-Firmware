65 static void ADS1256_LowLevel_Init(void) {
66  GPIO_InitTypeDef GPIO_InitStructure;
67 
68  /* Enable the SPI clock */
69  ADS1256_SPI_CLK_INIT(ADS1256_SPI_CLK, ENABLE);
70 
71  /* Enable GPIO clocks */
72  RCC_AHB1PeriphClockCmd(ADS1256_SPI_SCK_GPIO_CLK | ADS1256_SPI_MISO_GPIO_CLK |
73  ADS1256_SPI_MOSI_GPIO_CLK | ADS1256_CS_GPIO_CLK, ENABLE);
74 
75  /* Connect SPI pins to AF5 */
76  GPIO_PinAFConfig(ADS1256_SPI_SCK_GPIO_PORT, ADS1256_SPI_SCK_SOURCE, ADS1256_SPI_SCK_AF);
77  GPIO_PinAFConfig(ADS1256_SPI_MISO_GPIO_PORT, ADS1256_SPI_MISO_SOURCE, ADS1256_SPI_MISO_AF);
78  GPIO_PinAFConfig(ADS1256_SPI_MOSI_GPIO_PORT, ADS1256_SPI_MOSI_SOURCE, ADS1256_SPI_MOSI_AF);
79 
80  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
81  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_2MHz;
82  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
83  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_DOWN;
84 
85  /* SPI SCK pin configuration */
86  GPIO_InitStructure.GPIO_Pin = ADS1256_SPI_SCK_PIN;
87  GPIO_Init(ADS1256_SPI_SCK_GPIO_PORT, &GPIO_InitStructure);
88 
89  /* SPI MOSI pin configuration */
90  GPIO_InitStructure.GPIO_Pin = ADS1256_SPI_MOSI_PIN;
91  GPIO_Init(ADS1256_SPI_MOSI_GPIO_PORT, &GPIO_InitStructure);
92 
93  /* SPI MISO pin configuration */
94  GPIO_InitStructure.GPIO_Pin = ADS1256_SPI_MISO_PIN;
95  GPIO_Init(ADS1256_SPI_MISO_GPIO_PORT, &GPIO_InitStructure);
96 
97  /* Configure ADS1256 CS pin in output pushpull mode */
98  GPIO_InitStructure.GPIO_Pin = ADS1256_CS_PIN;
99  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
100  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
101  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
102  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
103  GPIO_Init(ADS1256_CS_GPIO_PORT, &GPIO_InitStructure);
104 }

111 static void ADS1256_LowLevel_DeInit(void) {
112  GPIO_InitTypeDef GPIO_InitStructure;
113 
114  /* Disable the ADS1256_SPI */
115  SPI_Cmd(ADS1256_SPI, DISABLE);
116 
117  /* DeInitializes the ADS1256_SPI */

123  /* Configure all pins used by the SPI as input floating */

150 void ADS1256_SPI_Init(void) {
151 #ifdef ADS1256_SPI_DEBUG
152  printf("[ADS1256] Initializing SPI peripherals for the ADS1256.\n\r");
153 #endif
154  SPI_InitTypeDef SPI_InitStructure;
155  ADS1256_LowLevel_Init();
156 
157  /* Deselect the ADS1256: Chip Select high */
158  ADS1256_CS_HIGH();
159 
160  /* SPI configuration */
161  SPI_InitStructure.SPI_Direction = SPI_Direction_2Lines_FullDuplex;
162  SPI_InitStructure.SPI_Mode = SPI_Mode_Master;
163  SPI_InitStructure.SPI_DataSize = SPI_DataSize_8b;
164  SPI_InitStructure.SPI_CPOL = SPI_CPOL_Low;
165  SPI_InitStructure.SPI_CPHA = SPI_CPHA_2Edge;
166  SPI_InitStructure.SPI_NSS = SPI_NSS_Soft;
167  SPI_InitStructure.SPI_BaudRatePrescaler = SPI_BaudRatePrescaler_8;
168  SPI_InitStructure.SPI_FirstBit = SPI_FirstBit_MSB;
169  SPI_InitStructure.SPI_CRCPolynomial = 7;
170 
171  SPI_Init(ADS1256_SPI, &SPI_InitStructure);
172 
173  /* Enable the ADC1256 SPI */
174  SPI_Cmd(ADS1256_SPI, ENABLE);
175 }

255 uint8_t ADS1256_SendByte(uint8_t data) {
256 #ifdef ADS1256_SPI_DEBUG
257  printf("[ADS1256] Sending byte: 0x%02X\n\r", data);
258 #endif
259  /* Send byte through the SPI peripheral */
260  SPI_I2S_SendData(ADS1256_SPI, data);
261 
262  /* Loop while DR register in not empty */
263  while (SPI_I2S_GetFlagStatus(ADS1256_SPI, SPI_I2S_FLAG_TXE) == RESET);
264 
265  /* Wait to receive a byte */
266  while (SPI_I2S_GetFlagStatus(ADS1256_SPI, SPI_I2S_FLAG_RXNE) == RESET);
267 
268  /* Return the byte read from the SPI bus */
269  return SPI_I2S_ReceiveData(ADS1256_SPI);
270 }
