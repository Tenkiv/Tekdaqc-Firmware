125 uint32_t GetLSIFrequency() {
126  NVIC_InitTypeDef NVIC_InitStructure;
127  TIM_ICInitTypeDef TIM_ICInitStructure;
128  RCC_ClocksTypeDef RCC_ClockFreq;
129 
130  /* Enable the LSI oscillator */
131  RCC_LSICmd(ENABLE);
132 
133  /* Wait till LSI is ready */
134  while (RCC_GetFlagStatus(RCC_FLAG_LSIRDY) == RESET) {
135  /* Do nothing */
136  }
137 
138  /* TIM5 configuration */
139  /* Enable TIM5 clock */
140  RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM5, ENABLE);
141 
142  /* Connect internally the TIM5_CH4 Input Capture to the LSI clock output */
143  TIM_RemapConfig(TIM5, TIM5_LSI);
144 
145  /* Configure TIM5 prescaler */
146  TIM_PrescalerConfig(TIM5, 0, TIM_PSCReloadMode_Immediate);
147 
148  /* TIM5 configuration: Input Capture mode
149  The LSI oscillator is connected to TIM5 CH4
150  The Rising edge is used as active edge,
151  The TIM5 CCR4 is used to compute the frequency value */
152  TIM_ICInitStructure.TIM_Channel = TIM_Channel_4;
153  TIM_ICInitStructure.TIM_ICPolarity = TIM_ICPolarity_Rising;
154  TIM_ICInitStructure.TIM_ICSelection = TIM_ICSelection_DirectTI;
155  TIM_ICInitStructure.TIM_ICPrescaler = TIM_ICPSC_DIV8;
156  TIM_ICInitStructure.TIM_ICFilter = 0;
157  TIM_ICInit(TIM5, &TIM_ICInitStructure);
158 
159  /* Enable TIM5 Interrupt channel */
160  NVIC_InitStructure.NVIC_IRQChannel = TIM5_IRQn;
161  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 0;
162  NVIC_InitStructure.NVIC_IRQChannelSubPriority = 0;
163  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
164  NVIC_Init(&NVIC_InitStructure);
165 
166  /* Enable TIM5 counter */
167  TIM_Cmd(TIM5, ENABLE);
168 
169  /* Reset the flags */
170  TIM5 ->SR = 0;
171 
172  /* Enable the CC4 Interrupt Request */
173  TIM_ITConfig(TIM5, TIM_IT_CC4, ENABLE);
174 
175  /* Wait until the TIM5 get 2 LSI edges (refer to TIM5_IRQHandler() in stm32f4xx_it.c file) */
176  while (LSICaptureNumber != 2) {
177  /* Do Nothing */
178  }
179  /* Deinitialize the TIM5 peripheral registers to their default reset values */
180  TIM_DeInit(TIM5);
181 
182  /* Compute the LSI frequency, depending on TIM5 input clock frequency (PCLK1) */
183  /* Get SYSCLK, HCLK and PCLKx frequency */
184  RCC_GetClocksFreq(&RCC_ClockFreq);
185 
186  /* Get PCLK1 prescaler */
187  if ((RCC ->CFGR & RCC_CFGR_PPRE1)== 0) {
188  /* PCLK1 prescaler equal to 1 => TIMCLK = PCLK1 */
189  return ((RCC_ClockFreq.PCLK1_Frequency / LSIPeriodValue) * 8);
190  } else {
191  /* PCLK1 prescaler different from 1 => TIMCLK = 2 * PCLK1 */
192  return (((2 * RCC_ClockFreq.PCLK1_Frequency) / LSIPeriodValue) * 8);
193  }
194 }

209 void Watchdog_Init(void) {
210  /* Get the LSI frequency from TIM5 */
211  LsiFreq = GetLSIFrequency();
212  printf("LSI Frequency: %" PRIu32 "\n", LsiFreq);
213 
214  /* Enable write access to IWDG_PR and IWDG_RLR registers */
215  IWDG_WriteAccessCmd(IWDG_WriteAccess_Enable);
216 
217  /* IWDG counter clock: LSI/32 */
218  IWDG_SetPrescaler(IWDG_Prescaler_64);
219 
220  /* Set counter reload value to obtain 1s IWDG TimeOut.
221  IWDG counter clock Frequency = LsiFreq/32
222  Counter Reload Value = 250ms/IWDG counter clock period
223  = 0.25s / (32/LsiFreq)
224  = LsiFreq/(32 * 4)
225  = LsiFreq/128
226  */
227  IWDG_SetReload(LsiFreq/16);
228 
229  /* Reload IWDG counter */
230  IWDG_ReloadCounter();
231 
232  /* Enable IWDG (the LSI oscillator will be enabled by hardware) */
233  IWDG_Enable();
234 }

324 void COMInit(COM_TypeDef COM, USART_InitTypeDef* USART_InitStruct) {
325  if (COM == COM1 || COM == COM2) {
326  GPIO_InitTypeDef GPIO_InitStructure;
327 
328  /* Enable GPIO clock */
329  RCC_AHB1PeriphClockCmd(COM_TX_PORT_CLK[COM] | COM_RX_PORT_CLK[COM], ENABLE);
330 
331  /* Enable UART clock */
332  RCC_APB1PeriphClockCmd(COM_USART_CLK[COM], ENABLE);
333 
334  /* Connect PXx to USARTx_Tx*/
335  GPIO_PinAFConfig(COM_TX_PORT[COM], COM_TX_PIN_SOURCE[COM], COM_TX_AF[COM]);
336 
337  /* Connect PXx to USARTx_Rx*/
338  GPIO_PinAFConfig(COM_RX_PORT[COM], COM_RX_PIN_SOURCE[COM], COM_RX_AF[COM]);
339 
340  /* Configure USART Tx as alternate function */
341  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
342  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
343  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
344 
345  GPIO_InitStructure.GPIO_Pin = COM_TX_PIN[COM];
346  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
347  GPIO_Init(COM_TX_PORT[COM], &GPIO_InitStructure);
348 
349  /* Configure USART Rx as alternate function */
350  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
351  GPIO_InitStructure.GPIO_Pin = COM_RX_PIN[COM];
352  GPIO_Init(COM_RX_PORT[COM], &GPIO_InitStructure);
353 
354  /* USART configuration */
355  USART_Init(COM_USART[COM], USART_InitStruct);
356 
357  /* Enable USART */
358  USART_Cmd(COM_USART[COM], ENABLE);
359  }
