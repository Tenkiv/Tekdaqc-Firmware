46 void TekDAQC_CheckStatus(void) {
47  /* Check on the digital outputs */
48  if (CheckDigitalOutputStatus() == true) {
49  if (TelnetIsConnected() == true) {
50  /* A fault was detected */
51  uint8_t n = snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "\n\r*DIGITAL OUTPUT FAULT DETECTED!*\n\r");
52  if (n > 0) {
53  TelnetWriteString(TOSTRING_BUFFER);
54  } else {
55 #ifdef DO_STATE_MACHINE_DEBUG
56  printf("[DO STATE MACHINE] Failed to write fault warning for digital output.\n\r");
57 #endif
58  }
59  }
60  }
