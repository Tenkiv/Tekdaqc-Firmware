309 Tekdaqc_Function_Error_t ListDigitalInputs(void) {
310  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
311  uint8_t n = snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "\n\r----------\n\rAdded Digital Inputs\n\r----------\n\r");
312  if (n <= 0) {
313 #ifdef DIGITALINPUT_DEBUG
314  printf("Failed to write header of digital input list.\n\r");
315 #endif
316  retval = ERR_DIN_FAILED_WRITE;
317  } else {
318  if (writer != 0) {
319  writer(TOSTRING_BUFFER);
320  }
321  Digital_Input_t input;
322  for (uint_fast8_t i = 0U; i < NUM_DIGITAL_INPUTS; ++i) {
323  input = Ext_DInputs[i];
324  if (input.added == CHANNEL_ADDED) {
325  /* This input has been added */
326  n = snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "\tPhysical Input %" PRIi8 ":\n\r\t\tName: %s\n\r", input.input,
327  input.name);
328  if (n <= 0) {
329 #ifdef DIGITALINPUT_DEBUG
330  printf("Failed to write an digital input to the list.\n\r");
331 #endif
332  retval = ERR_DIN_FAILED_WRITE;
333  break;
334  } else {
335  if (writer != 0) {
336  writer(TOSTRING_BUFFER);
337  }
338  }
339  }
340  }
341  }
342  return retval;
343 }

354 Tekdaqc_Function_Error_t CreateDigitalInput(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], int count) {
355  Tekdaqc_Function_Error_t retval = ERR_FUNCTION_OK;
356  char* param;
357  char* testPtr = NULL;
358  int8_t index = -1;
359  uint8_t input = NULL_CHANNEL; /* The physical input */
360  uint8_t in = NULL_CHANNEL;
361  char name[MAX_DIGITAL_INPUT_NAME_LENGTH]; /* The name */
362  strcpy(name, "NONE");
363  for (int i = 0; i < NUM_ADD_DIGITAL_INPUT_PARAMS; ++i) {
364  index = GetIndexOfArgument(keys, ADD_DIGITAL_INPUT_PARAMS[i], count);
365  if (index >= 0) { /* We found the key in the list */
366  param = values[index]; /* We use the discovered index for this key */
367  switch (i) { /* Switch on the key not position in arguments list */
368  case 0U: /* INPUT key */
369  in = (uint8_t) strtol(param, &testPtr, 10);
370  if (testPtr == param) {
371  retval = ERR_DIN_PARSE_ERROR;
372  } else {
373  if (in >= 0U && in <= NUM_DIGITAL_INPUTS) {
374  /* A valid input number */
375  input = in;
376  } else {
377  /* Input number out of range */
378 #ifdef DIGITALINPUT_DEBUG
379  printf("[Digital Input] The requested input number is invalid.\n\r");
380 #endif
381  retval = ERR_DIN_INPUT_OUTOFRANGE;
382  }
383  }
384  break;
385  case 1U: /* NAME key */
386  strcpy(name, param);
387  break;
388  default:
389  /* Return an error */
390  retval = ERR_DIN_PARSE_ERROR;
391  }
392  } else if (i == 1U) {
393  /* The NAME key is not strictly required, apply the defaults */
394  continue;
395  } else {
396  /* Somehow an error happened */
397 #ifdef DIGITALINPUT_DEBUG
398  printf("[Digital Input] Unable to locate required key: %s\n\r", ADD_DIGITAL_INPUT_PARAMS[i]);
399 #endif
400  retval = ERR_DIN_PARSE_MISSING_KEY; /* Failed to locate a key */
401  }
402  if (retval != ERR_FUNCTION_OK) {
403  break;
404  }
405  }
406  if (retval == ERR_FUNCTION_OK) {
407  if (input != NULL_CHANNEL ) {
408  Digital_Input_t* dig_input = GetDigitalInputByNumber(input);
409  if (dig_input != NULL ) {
410  if (dig_input->added == CHANNEL_NOTADDED) {
411  dig_input->input = input;
412  strcpy(dig_input->name, name);
413  dig_input->level = LOGIC_LOW;
414  dig_input->timestamp = 0U;
415  retval = AddDigitalInput(dig_input);
416  } else {
417  retval = ERR_DIN_INPUT_EXISTS;
418  }
419  } else {
420  /* The input could not be found */
421  return ERR_DIN_INPUT_NOT_FOUND;
422  }
423  } else {
424  /* A valid input was never specified */
425  retval = ERR_DIN_INPUT_UNSPECIFIED;
426  }
427  }
428  return retval;
429 }
