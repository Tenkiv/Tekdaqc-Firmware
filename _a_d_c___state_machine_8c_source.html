196 static void ADC_Machine_Service_Calibrating(void) {
197  static ADS1256_PGA_t gains[] = { ADS1256_PGAx1, ADS1256_PGAx2, ADS1256_PGAx4, ADS1256_PGAx8, ADS1256_PGAx16, ADS1256_PGAx32,
198  ADS1256_PGAx64 };
199  static ADS1256_SPS_t rates[] = { ADS1256_SPS_30000, ADS1256_SPS_15000, ADS1256_SPS_7500, ADS1256_SPS_3750, ADS1256_SPS_2000,
200  ADS1256_SPS_1000, ADS1256_SPS_500, ADS1256_SPS_100, ADS1256_SPS_60, ADS1256_SPS_50, ADS1256_SPS_30, ADS1256_SPS_25,
201  ADS1256_SPS_15, ADS1256_SPS_10, ADS1256_SPS_5, ADS1256_SPS_2_5 };
202  static ADS1256_BUFFER_t buffers[] = { ADS1256_BUFFER_ENABLED, ADS1256_BUFFER_DISABLED };
203 
204  if (calibrationState.finished == false) {
205  if (calibrationState.gain_index < NUM_PGA_SETTINGS) {
206  if (calibrationState.rate_index < NUM_SAMPLE_RATES) {
207  if (calibrationState.buffer_index < NUM_BUFFER_SETTINGS) {
208  ADS1256_SetInputBufferSetting(buffers[calibrationState.buffer_index]);
209  ADS1256_SetDataRate(rates[calibrationState.rate_index]);
210  ADS1256_SetPGASetting(gains[calibrationState.gain_index]);
211  ADS1256_CalibrateSelf();
212  Tekdaqc_SetBaseGainCalibration(ADS1256_GetGainCalSetting(), rates[calibrationState.rate_index],
213  gains[calibrationState.gain_index], buffers[calibrationState.buffer_index]);
214  ADS1256_CalibrateSystem_Offset();
215  Tekdaqc_SetOffsetCalibration(ADS1256_GetOffsetCalSetting(), rates[calibrationState.rate_index],
216  gains[calibrationState.gain_index], buffers[calibrationState.buffer_index++]);
217  calibrationState.finished_count++;
218  } else {
219  calibrationState.buffer_index = 0;
220  calibrationState.rate_index++;
221  }
222  } else {
223  calibrationState.rate_index = 0;
224  calibrationState.gain_index++;
225  }
226  } else {
227  calibrationState.finished = true;
228  }
229 #ifdef ADC_STATE_MACHINE_DEBUG
230  static uint32_t total = NUM_PGA_SETTINGS * NUM_SAMPLE_RATES * NUM_BUFFER_SETTINGS;
231  printf("[ADC STATE MACHINE] Calibration progress: %f%.\n\r", ((float) calibrationState.finished_count/total)*100.0f);
232 #endif
233  } else {
234  /* The calibration is complete */
235  ADC_Machine_Idle();
236  }
237 }

259 static void ADC_Machine_Service_Idle(void) {
260  /* If a temperature reading is ready, update the board temperature */
261  if (ADS1256_IsDataReady(false)) {
262  Analog_Input_t* input = GetAnalogInputByNumber(IN_COLD_JUNCTION);
263  waitingOnTemp = false;
264  /* We need to read it */
265  input->values[input->bufferWriteIdx] = ADS1256_GetMeasurement();
266  /* Update temperature */
267  updateBoardTemperature(input, input->values[input->bufferWriteIdx]);
268  ++(input->bufferWriteIdx);
269 #ifdef BOARD_TEMPERATURE_DEBUG
270  printf("[ADC STATE MACHINE] Cold junction temperature sample is complete.\n\r");
271 #endif
272  /* Check the buffer positions for errors/roll over */
273  if (((input->bufferWriteIdx - 1U) == input->bufferReadIdx) && ((input->bufferWriteIdx - 1U) > 0U)) {
274  /* This means we wrote over some data */
275 #ifdef BOARD_TEMPERATURE_DEBUG
276  printf("[ADC STATE MACHINE] Cold junction sampling overwrote data before it could be read.\n\r");
277 #endif
278  TelnetWriteErrorMessage("Cold junction sampling overwrote data before it could be read.");
279  }
280  if (input->bufferWriteIdx == ANALOG_INPUT_BUFFER_SIZE) {
281  /* We have reached the end of the buffer and need to start writing to the beginning */
282  input->bufferWriteIdx = 0U;
283  }
284  }

293 static void ADC_Machine_Service_Sampling(void) {
294  /* Check if data is ready */
295  if (ADS1256_IsDataReady(false)) {
296  /* We need to read it */
297  uint8_t writeIndex = samplingInputs[currentSamplingInput]->bufferWriteIdx;
298  samplingInputs[currentSamplingInput]->values[writeIndex] = ADS1256_GetMeasurement();
299  samplingInputs[currentSamplingInput]->bufferWriteIdx = (writeIndex + 1) % ANALOG_INPUT_BUFFER_SIZE;
300  if (samplingInputs[currentSamplingInput]->bufferWriteIdx == samplingInputs[currentSamplingInput]->bufferReadIdx) {
301  /* Check the buffer positions for errors/roll over */
302 #ifdef ADC_STATE_MACHINE_DEBUG
303  printf("[ADC STATE MACHINE] Analog sampling overwrote data before it could be read.\n\r");
304 #endif
305  TelnetWriteErrorMessage("Analog sampling overwrote data before it could be read.");
306  /* Buffer is full, overwrite */
307  samplingInputs[currentSamplingInput]->bufferReadIdx = (samplingInputs[currentSamplingInput]->bufferReadIdx + 1) % ANALOG_INPUT_BUFFER_SIZE;
308  }
309 
310  ADS1256_Sync(true); /*Halt conversion */
311  /* Select the next input */
312 
313  if (numberSamplingInputs > 1) {
314  Analog_Input_t* input = NULL;
315  Analog_Input_t* current = samplingInputs[currentSamplingInput];
316  while (input == NULL || input->added == CHANNEL_NOTADDED) { /* Keep searching until we find a non-null input */
317  ++currentSamplingInput;
318  if (currentSamplingInput == NUM_ANALOG_INPUTS) {
319  /* We have reached the end of a set */
320  currentSamplingInput = 0U; /* Reset to the start */
321 #ifdef ADC_STATE_MACHINE_DEBUG
322  printf("[ADC STATE MACHINE] Sample %" PRIi32 " of %" PRIi32 " is complete.\n\r", SampleCurrent + 1, SampleTotal);
323 #endif
324  ++SampleCurrent; /* Increment the sample counter */
325  }
326  input = samplingInputs[currentSamplingInput];
327  }
328  if (input != current) { /* Prevent unnecessary external muxings */
329  SelectAnalogInput(input); /* Select the input */
330  } else {
331 #ifdef ADC_STATE_MACHINE_DEBUG
332  printf("[ADC STATE MACHINE] Multi-channel sampling tried to select the currently selected input. Ignoring...\n\r");
333 #endif
334  ADS1256_Wakeup(); /* Begin the next sample */
335  /* Save the real time clock entry for the sample */
336  samplingInputs[currentSamplingInput]->timestamps[samplingInputs[currentSamplingInput]->bufferWriteIdx] = GetLocalTime();
337  }
338  } else {
339  /* We are single channel sampling */
340 #ifdef ADC_STATE_MACHINE_DEBUG
341  printf("[ADC STATE MACHINE] Sample %" PRIi32 " of %" PRIi32 " is complete.\n\r", SampleCurrent + 1, SampleTotal);
342 #endif
343  ++SampleCurrent; /* Increment the sample count */
344  ADS1256_Wakeup(); /* Begin the next sample */
345  /* Save the real time clock entry for the sample */
346  samplingInputs[currentSamplingInput]->timestamps[samplingInputs[currentSamplingInput]->bufferWriteIdx] = GetLocalTime();
347  }
348  if ((SampleCurrent == SampleTotal) && !((numberSamplingInputs > 1) && SampleCurrent == 0)) {
349  /* The equality check is because we incremented already */
350  /* We are done sampling, write out any remaining data and return to idle state */
351  WriteAnalogInput(samplingInputs[currentSamplingInput]);
352  ADC_Machine_Idle();
353 #ifdef ADC_STATE_MACHINE_DEBUG
354  printf("[ADC STATE MACHINE] Channel sampling is complete.\n\r");
355 #endif
356  TelnetWriteStatusMessage("ADC Channel sampling completed.");
357  /* We are done, return */
358  CompletedADCSampling();
359  return;
360  }
361  } else {
362  /* There is nothing to read yet, write some data */
363  /* TODO: This needs to handle other inputs most likely */
364  WriteAnalogInput(samplingInputs[currentSamplingInput]);
365  }
366 }

375 static void ADC_Machine_Service_Muxing(void) {
376  Analog_Input_t* input = GetAnalogInputByNumber(IN_COLD_JUNCTION);
377  if (waitingOnTemp == false) {
378  /* We need to begin a temperature sample */
379  ADS1256_Sync(true);
380  SelectColdJunctionInput();
381  /* Set sampling parameters */
382  ADS1256_SetDataRate(input->rate);
383  ADS1256_SetPGASetting(input->gain);
384  ADS1256_SetInputBufferSetting(input->buffer);
385  ApplyCalibrationParameters(input);
386  ADS1256_Wakeup();
387  input->timestamps[input->bufferWriteIdx] = GetLocalTime();
388  waitingOnTemp = true;
389  } else {
390  /* We are waiting for a temperature sample to complete */
391  if (ADS1256_IsDataReady(false)) {
392  /* We need to read it */
393  uint8_t writeIndex = input->bufferWriteIdx;
394  input->values[writeIndex] = ADS1256_GetMeasurement();
395  /* Update temperature */
396  updateBoardTemperature(input, input->values[writeIndex]);
397  input->bufferWriteIdx = (writeIndex + 1) % ANALOG_INPUT_BUFFER_SIZE;
398  if (input->bufferWriteIdx == input->bufferReadIdx) {
399  /* Check the buffer positions for errors/roll over */
400 /*#ifdef ADC_STATE_MACHINE_DEBUG
401  printf("[ADC STATE MACHINE] Analog sampling overwrote data before it could be read.\n\r");
402 #endif
403  TelnetWriteErrorMessage("Analog sampling overwrote data before it could be read.");*/
404  /* Buffer is full, overwrite */
405  input->bufferReadIdx = (input->bufferReadIdx + 1) % ANALOG_INPUT_BUFFER_SIZE;
406  }
407 
408 #ifdef BOARD_TEMPERATURE_DEBUG
409  printf("[ADC STATE MACHINE] Cold junction temperature sample is complete.\n\r");
410 #endif
411  ADS1256_Sync(false); /*Halt conversion */
412  }
413  if ((PreviousState == ADC_CHANNEL_SAMPLING) && (isExternalMuxingComplete() == true)) {
414  /* We need to select external inputs on the internal multiplexer */
415  ResetSelectedInput();
416  /* We need to begin the next conversion */
417  BeginNextConversion(samplingInputs[currentSamplingInput]);
418  CurrentState = PreviousState; /* Return the state to its previous value */
419  } else if (((PreviousState == ADC_IDLE) || (PreviousState == ADC_CALIBRATING) || (PreviousState == ADC_GAIN_CALIBRATING))
420  && (isExternalMuxingComplete() == true)) {
421  CurrentState = PreviousState;
422  }
423  }
424 }

559 void ADC_Machine_Service(void) {
560  /* Determine the state */
561  switch (CurrentState) {
562  case ADC_UNINITIALIZED:
563  ADC_Machine_Init();
564  break;
565  case ADC_INITIALIZED:
566  ADC_Machine_Idle();
567  break;
568  case ADC_CALIBRATING:
569  ADC_Machine_Service_Calibrating();
570  break;
571  case ADC_GAIN_CALIBRATING:
572  ADC_Machine_Service_GainCalibrating();
573  break;
574  case ADC_IDLE:
575  if (isFirstIdle == true) {
576  isFirstIdle = false;
577  PerformSystemCalibration(); /* Load the offset calibration table */
578  } else {
579  ADC_Machine_Service_Idle();
580  }
581  break;
582  case ADC_CHANNEL_SAMPLING:
583  ADC_Machine_Service_Sampling();
584  break;
585  case ADC_RESET:
586  ADS1256_Full_Reset();
587  SampleCurrent = 0U;
588  SampleTotal = 0U;
589  samplingInputs = NULL;
590  numberSamplingInputs = 0U;
591  currentSamplingInput = NULL_CHANNEL;
592  ADC_Machine_Idle();
593  break;
594  case ADC_EXTERNAL_MUXING:
595  ADC_Machine_Service_Muxing();
596  break;
597  default:
598  /* TODO: Throw an error */
599  break;
600  }
601 }

656 void ADC_Machine_Input_Sample(Analog_Input_t** inputs, uint32_t count, bool singleChannel) {
657  /* We use explicit checking here in case other state are added later */
658  if ((CurrentState != ADC_IDLE)) {
659 #ifdef ADC_STATE_MACHINE_DEBUG
660  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state from %s\n\r",
661  ADCMachine_StringFromState(CurrentState));
662 #endif
663  return;
664  }
665  /* Validate the inputs list */
666  if (inputs == NULL ) {
667 #ifdef ADC_STATE_MACHINE_DEBUG
668  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input list. Ignoring...\n\r");
669 #endif
670  return;
671  }
672 
673  /* Save sample count */
674  SampleCurrent = 0U;
675  SampleTotal = count;
676 
677  if (singleChannel == true) {
678  /* Validate the input(s) */
679  if (inputs[0] == NULL ) {
680 #ifdef ADC_STATE_MACHINE_DEBUG
681  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input. Ignoring...\n\r");
682 #endif
683  return;
684  }
685  if (inputs[0]->added == CHANNEL_NOTADDED) {
686 #ifdef ADC_STATE_MACHINE_DEBUG
687  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with an un-added analog input. Ignoring...\n\r");
688 #endif
689  return;
690  }
691  /* Select input */
692  currentSamplingInput = 0;
693  numberSamplingInputs = 1;
694  } else {
695  /* Validate the input(s) */
696  uint8_t i = 0U;
697  for (; i < NUM_ANALOG_INPUTS; ++i) {
698  if (inputs[i] != NULL && inputs[i]->added == CHANNEL_ADDED) {
699  break;
700  }
701  }
702  if (i >= NUM_ANALOG_INPUTS) {
703 #ifdef ADC_STATE_MACHINE_DEBUG
704  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input. Ignoring...\n\r");
705 #endif
706  return;
707  }
708  /* Select input */
709  currentSamplingInput = i; /* Use i here so we don't loose any work from the search earlier */
710  numberSamplingInputs = NUM_ANALOG_INPUTS;
711  }
712 
713  samplingInputs = inputs;
714  Analog_Input_t* input = samplingInputs[currentSamplingInput];
715  CurrentState = ADC_CHANNEL_SAMPLING;
716  SelectAnalogInput(input);
717 
718  if (CurrentState == ADC_CHANNEL_SAMPLING) {
719  /* Set sampling parameters */
720  ADS1256_SetDataRate(input->rate);
721  ADS1256_SetPGASetting(input->gain);
722  ADS1256_SetInputBufferSetting(input->buffer);
723  ApplyCalibrationParameters(input);
724  /* Begin sampling */
725  ADS1256_Sync(false);
726  ADS1256_Wakeup(); /* Start Sampling */
727  /* Save the real time clock entry for the sample */
728  input->timestamps[input->bufferWriteIdx] = GetLocalTime();
729  } else {
730  /* We entered the ADC_MUXING state */
731  ADS1256_Sync(false); /* TODO: Is this necessary? It may cause temperature fluctuations */
732  }
733 }
