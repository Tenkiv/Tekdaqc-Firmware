196 static void ADC_Machine_Service_Calibrating(void) {
197  static ADS1256_PGA_t gains[] = { ADS1256_PGAx1, ADS1256_PGAx2, ADS1256_PGAx4, ADS1256_PGAx8, ADS1256_PGAx16, ADS1256_PGAx32,
198  ADS1256_PGAx64 };
199  static ADS1256_SPS_t rates[] = { ADS1256_SPS_30000, ADS1256_SPS_15000, ADS1256_SPS_7500, ADS1256_SPS_3750, ADS1256_SPS_2000,
200  ADS1256_SPS_1000, ADS1256_SPS_500, ADS1256_SPS_100, ADS1256_SPS_60, ADS1256_SPS_50, ADS1256_SPS_30, ADS1256_SPS_25,
201  ADS1256_SPS_15, ADS1256_SPS_10, ADS1256_SPS_5, ADS1256_SPS_2_5 };
202  static ADS1256_BUFFER_t buffers[] = { ADS1256_BUFFER_ENABLED, ADS1256_BUFFER_DISABLED };
203 
204  if (calibrationState.finished == false) {
205  if (calibrationState.gain_index < NUM_PGA_SETTINGS) {
206  if (calibrationState.rate_index < NUM_SAMPLE_RATES) {
207  if (calibrationState.buffer_index < NUM_BUFFER_SETTINGS) {
208  ADS1256_SetInputBufferSetting(buffers[calibrationState.buffer_index]);
209  ADS1256_SetDataRate(rates[calibrationState.rate_index]);
210  ADS1256_SetPGASetting(gains[calibrationState.gain_index]);
211  ADS1256_CalibrateSelf();
212  Tekdaqc_SetBaseGainCalibration(ADS1256_GetGainCalSetting(), rates[calibrationState.rate_index],
213  gains[calibrationState.gain_index], buffers[calibrationState.buffer_index]);
214  ADS1256_CalibrateSystem_Offset();
215  Tekdaqc_SetOffsetCalibration(ADS1256_GetOffsetCalSetting(), rates[calibrationState.rate_index],
216  gains[calibrationState.gain_index], buffers[calibrationState.buffer_index++]);
217  calibrationState.finished_count++;
218  } else {
219  calibrationState.buffer_index = 0;
220  calibrationState.rate_index++;
221  }
222  } else {
223  calibrationState.rate_index = 0;
224  calibrationState.gain_index++;
225  }
226  } else {
227  calibrationState.finished = true;
228  }
229 #ifdef ADC_STATE_MACHINE_DEBUG
230  static uint32_t total = NUM_PGA_SETTINGS * NUM_SAMPLE_RATES * NUM_BUFFER_SETTINGS;
231  snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "[ADC STATE MACHINE] Calibration progress: %f\%.\n\r",
232  ((float) calibrationState.finished_count / total) * 100.0f);
233  printf("%s", TOSTRING_BUFFER);
234  TelnetWriteStatusMessage(TOSTRING_BUFFER);
235 #endif
236  } else {
237  /* The calibration is complete */
238  printf("[ADC STATE MACHINE] Calibration Completed.\n\r");
239  TelnetWriteStatusMessage("[ADC STATE MACHINE] Calibration Completed.\n\r");
240  ADC_Machine_Idle();
241  }
242 }

264 static void ADC_Machine_Service_Idle(void) {
265  /* If a temperature reading is ready, update the board temperature */
266  if (ADS1256_IsDataReady(false)) {
267  Analog_Input_t* input = GetAnalogInputByNumber(IN_COLD_JUNCTION);
268  ADS1256_Sync(true);
269  waitingOnTemp = false;
270  /* We need to read it */
271  input->values[input->bufferWriteIdx] = ADS1256_GetMeasurement();
272  /* Update temperature */
273  updateBoardTemperature(input, input->values[input->bufferWriteIdx]);
274  ++(input->bufferWriteIdx);
275  input->timestamps[input->bufferWriteIdx] = GetLocalTime();
276  ADS1256_Wakeup();
277 #ifdef BOARD_TEMPERATURE_DEBUG
278  printf("[ADC STATE MACHINE] Cold junction temperature sample is complete.\n\r");
279 #endif
280  /* Check the buffer positions for errors/roll over */
281  if (input->bufferWriteIdx == input->bufferReadIdx) {
282  /* Check the buffer positions for errors/roll over */
283 #ifdef BOARD_TEMPERATURE_DEBUG
284  printf("[ADC STATE MACHINE] Cold junction sampling overwrote data before it could be read.\n\r");
285  TelnetWriteErrorMessage("[ADC STATE MACHINE] Cold junction sampling overwrote data before it could be read.\n\r");
286 #endif
287  /* Buffer is full, overwrite */
288  input->bufferReadIdx = (input->bufferReadIdx + 1) % ANALOG_INPUT_BUFFER_SIZE;
289  }
290  if (((input->bufferWriteIdx - 1U) == input->bufferReadIdx) && ((input->bufferWriteIdx - 1U) > 0U)) {
291  /* This means we wrote over some data */
292  }
293  if (input->bufferWriteIdx == ANALOG_INPUT_BUFFER_SIZE) {
294  /* We have reached the end of the buffer and need to start writing to the beginning */
295  input->bufferWriteIdx = 0U;
296  }
297  }
298 }

306 static void ADC_Machine_Service_Sampling(void) {
307  /* Check if data is ready */
308  if (ADS1256_IsDataReady(false)) {
309  /* We need to read it */
310  printf("Reading ADC Sample.\n\r");
311  ADS1256_PrintRegs();
312  uint8_t writeIndex = samplingInputs[currentSamplingInput]->bufferWriteIdx;
313  samplingInputs[currentSamplingInput]->values[writeIndex] = ADS1256_GetMeasurement();
314  samplingInputs[currentSamplingInput]->bufferWriteIdx = (writeIndex + 1) % ANALOG_INPUT_BUFFER_SIZE;
315  if (samplingInputs[currentSamplingInput]->bufferWriteIdx == samplingInputs[currentSamplingInput]->bufferReadIdx) {
316  /* Check the buffer positions for errors/roll over */
317 #ifdef ADC_STATE_MACHINE_DEBUG
318  printf("[ADC STATE MACHINE] Analog sampling overwrote data before it could be read.\n\r");
319 #endif
320  TelnetWriteErrorMessage("Analog sampling overwrote data before it could be read.");
321  /* Buffer is full, overwrite */
322  samplingInputs[currentSamplingInput]->bufferReadIdx = (samplingInputs[currentSamplingInput]->bufferReadIdx + 1)
323  % ANALOG_INPUT_BUFFER_SIZE;
324  }
325 
326  ADS1256_Sync(true); /*Halt conversion */
327  /* Select the next input */
328 
329  if (numberSamplingInputs > 1) {
330  Analog_Input_t* input = NULL;
331  Analog_Input_t* current = samplingInputs[currentSamplingInput];
332  while (input == NULL || input->added == CHANNEL_NOTADDED) { /* Keep searching until we find a non-null input */
333  ++currentSamplingInput;
334  if (currentSamplingInput == NUM_ANALOG_INPUTS) {
335  /* We have reached the end of a set */
336  currentSamplingInput = 0U; /* Reset to the start */
337 #ifdef ADC_STATE_MACHINE_DEBUG
338  printf("[ADC STATE MACHINE] Sample %" PRIi32 " of %" PRIi32 " is complete.\n\r", SampleCurrent + 1, SampleTotal);
339 #endif
340  ++SampleCurrent; /* Increment the sample counter */
341  }
342  input = samplingInputs[currentSamplingInput];
343  }
344  if (input != current) { /* Prevent unnecessary external muxings */
345  SelectAnalogInput(input, true); /* Select the input */
346  /* We are done sampling this input, write out any remaining data */
347  WriteAnalogInput(current);
348  } else {
349 #ifdef ADC_STATE_MACHINE_DEBUG
350  printf("[ADC STATE MACHINE] Multi-channel sampling tried to select the currently selected input. Ignoring...\n\r");
351 #endif
352  ADS1256_Wakeup(); /* Begin the next sample */
353  /* Save the real time clock entry for the sample */
354  samplingInputs[currentSamplingInput]->timestamps[samplingInputs[currentSamplingInput]->bufferWriteIdx] = GetLocalTime();
355  }
356  } else {
357  /* We are single channel sampling */
358 #ifdef ADC_STATE_MACHINE_DEBUG
359  snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "[ADC STATE MACHINE] Sample %" PRIi32 " of %" PRIi32 " is complete.\n\r",
360  SampleCurrent + 1, SampleTotal);
361  TelnetWriteStatusMessage(TOSTRING_BUFFER);
362 #endif
363  ++SampleCurrent; /* Increment the sample count */
364  ADS1256_Wakeup(); /* Begin the next sample */
365  /* Save the real time clock entry for the sample */
366  samplingInputs[currentSamplingInput]->timestamps[samplingInputs[currentSamplingInput]->bufferWriteIdx] = GetLocalTime();
367  }
368  if ((SampleCurrent == SampleTotal) && !((numberSamplingInputs > 1) && SampleCurrent == 0)) {
369  /* The equality check is because we incremented already */
370  /* We are done sampling, write out any remaining data and return to idle state */
371  WriteAnalogInput(samplingInputs[currentSamplingInput]);
372  ADC_Machine_Idle();
373 #ifdef ADC_STATE_MACHINE_DEBUG
374  printf("[ADC STATE MACHINE] Channel sampling is complete.\n\r");
375 #endif
376  TelnetWriteStatusMessage("ADC Channel sampling completed.");
377  /* We are done, return */
378  CompletedADCSampling();
379  return;
380  }
381  } else {
382  /* There is nothing to read yet, write some data */
383  /* TODO: This needs to handle other inputs most likely */
384  WriteAnalogInput(samplingInputs[currentSamplingInput]);
385  }
386 }

395 static void ADC_Machine_Service_Muxing(void) {
396  Analog_Input_t* input = GetAnalogInputByNumber(IN_COLD_JUNCTION);
397  if (waitingOnTemp == false) {
398  /* We need to begin a temperature sample */
399  ADS1256_Sync(true);
400  SelectColdJunctionInput();
401  Analog_Input_t* cold = GetAnalogInputByNumber(IN_COLD_JUNCTION);
402  BeginNextConversion(cold);
403  waitingOnTemp = true;
404  } else {
405  /* We are waiting for a temperature sample to complete */
406  if (ADS1256_IsDataReady(false)) {
407  /* We need to read it */
408  uint8_t writeIndex = input->bufferWriteIdx;
409  input->values[writeIndex] = ADS1256_GetMeasurement();
410  /* Update temperature */
411  updateBoardTemperature(input, input->values[writeIndex]);
412  input->bufferWriteIdx = (writeIndex + 1) % ANALOG_INPUT_BUFFER_SIZE;
413  if (input->bufferWriteIdx == input->bufferReadIdx) {
414  /* Check the buffer positions for errors/roll over */
415  /*#ifdef ADC_STATE_MACHINE_DEBUG
416  printf("[ADC STATE MACHINE] Analog sampling overwrote data before it could be read.\n\r");
417  #endif
418  TelnetWriteErrorMessage("Analog sampling overwrote data before it could be read.");*/
419  /* Buffer is full, overwrite */
420  input->bufferReadIdx = (input->bufferReadIdx + 1) % ANALOG_INPUT_BUFFER_SIZE;
421  }
422 
423 #ifdef BOARD_TEMPERATURE_DEBUG
424  printf("[ADC STATE MACHINE] Cold junction temperature sample is complete.\n\r");
425 #endif
426  ADS1256_Sync(false); /*Halt conversion */
427  }
428  if ((PreviousState == ADC_CHANNEL_SAMPLING) && (isExternalMuxingComplete() == true)) {
429  /* We need to select external inputs on the internal multiplexer */
430  ResetSelectedInput();
431  /* We need to begin the next conversion */
432  BeginNextConversion(samplingInputs[currentSamplingInput]);
433  CurrentState = PreviousState; /* Return the state to its previous value */
434  } else if (((PreviousState == ADC_IDLE) || (PreviousState == ADC_CALIBRATING) || (PreviousState == ADC_GAIN_CALIBRATING))
435  && (isExternalMuxingComplete() == true)) {
436  ResetSelectedInput();
437  CurrentState = PreviousState;
438  }
439  }
440 }

448 static void ApplyCalibrationParameters(Analog_Input_t* input) {
449  uint32_t offset_cal;
450  uint32_t gain_cal;
451  printf("Applying calibration parameters for input: %s\n\r", input->name);
452  if (input->physicalInput == IN_COLD_JUNCTION) {
453  offset_cal = Tekdaqc_GetColdJunctionOffsetCalibration();
454  gain_cal = Tekdaqc_GetColdJunctionGainCalibration();
455  } else {
456  offset_cal = Tekdaqc_GetOffsetCalibration(input->rate, input->gain, input->buffer);
457  gain_cal = Tekdaqc_GetGainCalibration(input->rate, input->gain, input->buffer, getBoardTemperature());
458  }
459  printf("Calibration params: Offset: 0x%" PRIx32 " Gain: 0x%" PRIx32 "\n\r", offset_cal, gain_cal);
460  ConvertCalibrationToBytes(scratch_bytes, offset_cal);
461  ADS1256_SetOffsetCalSetting(scratch_bytes);
462  ConvertCalibrationToBytes(scratch_bytes, gain_cal);
463  ADS1256_SetGainCalSetting(scratch_bytes);
464 }

590 void ADC_Machine_Service(void) {
591  /* Determine the state */
592  switch (CurrentState) {
593  case ADC_UNINITIALIZED:
594  ADC_Machine_Init();
595  break;
596  case ADC_INITIALIZED:
597  ADC_Machine_Idle();
598  break;
599  case ADC_CALIBRATING:
600  ADC_Machine_Service_Calibrating();
601  break;
602  case ADC_GAIN_CALIBRATING:
603  ADC_Machine_Service_GainCalibrating();
604  break;
605  case ADC_IDLE:
606  if (isFirstIdle == true) {
607  isFirstIdle = false;
608  PerformSystemCalibration(); /* Load the offset calibration table */
609  } else {
610  ADC_Machine_Service_Idle();
611  }
612  break;
613  case ADC_CHANNEL_SAMPLING:
614  ADC_Machine_Service_Sampling();
615  break;
616  case ADC_RESET:
617  ADS1256_Full_Reset();
618  SampleCurrent = 0U;
619  SampleTotal = 0U;
620  samplingInputs = NULL;
621  numberSamplingInputs = 0U;
622  currentSamplingInput = NULL_CHANNEL;
623  ADC_Machine_Idle();
624  break;
625  case ADC_EXTERNAL_MUXING:
626  ADC_Machine_Service_Muxing();
627  break;
628  default:
629  /* TODO: Throw an error */
630  break;
631  }
632 }

687 void ADC_Machine_Input_Sample(Analog_Input_t** inputs, uint32_t count, bool singleChannel) {
688  /* We use explicit checking here in case other state are added later */
689  if ((CurrentState != ADC_IDLE)) {
690 #ifdef ADC_STATE_MACHINE_DEBUG
691  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state from %s\n\r", ADCMachine_StringFromState(CurrentState));
692 #endif
693  return;
694  }
695  /* Validate the inputs list */
696  if (inputs == NULL) {
697 #ifdef ADC_STATE_MACHINE_DEBUG
698  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input list. Ignoring...\n\r");
699 #endif
700  return;
701  }
702 
703  /* Save sample count */
704  SampleCurrent = 0U;
705  SampleTotal = count;
706 
707  if (singleChannel == true) {
708  /* Validate the input(s) */
709  if (inputs[0] == NULL) {
710 #ifdef ADC_STATE_MACHINE_DEBUG
711  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input. Ignoring...\n\r");
712 #endif
713  return;
714  }
715  if (inputs[0]->added == CHANNEL_NOTADDED) {
716 #ifdef ADC_STATE_MACHINE_DEBUG
717  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with an un-added analog input. Ignoring...\n\r");
718 #endif
719  return;
720  }
721  /* Select input */
722  currentSamplingInput = 0;
723  numberSamplingInputs = 1;
724  } else {
725  /* Validate the input(s) */
726  uint8_t i = 0U;
727  for (; i < NUM_ANALOG_INPUTS; ++i) {
728  if (inputs[i] != NULL && inputs[i]->added == CHANNEL_ADDED) {
729  break;
730  }
731  }
732  if (i >= NUM_ANALOG_INPUTS) {
733 #ifdef ADC_STATE_MACHINE_DEBUG
734  printf("[ADC STATE MACHINE] Attempted to enter ADC_CHANNEL_SAMPLING state with a NULL analog input. Ignoring...\n\r");
735 #endif
736  return;
737  }
738  /* Select input */
739  currentSamplingInput = i; /* Use i here so we don't loose any work from the search earlier */
740  numberSamplingInputs = NUM_ANALOG_INPUTS;
741  }
742 
743  samplingInputs = inputs;
744  Analog_Input_t* input = samplingInputs[currentSamplingInput];
745  CurrentState = ADC_CHANNEL_SAMPLING;
746  SelectAnalogInput(input, true);
747 
748  if (CurrentState == ADC_CHANNEL_SAMPLING) {
749  /* Set sampling parameters */
750  ADS1256_SetDataRate(input->rate);
751  ADS1256_SetPGASetting(input->gain);
752  ADS1256_SetInputBufferSetting(input->buffer);
753  ApplyCalibrationParameters(input);
754  /* Begin sampling */
755  ADS1256_Sync(false);
756  ADS1256_Wakeup(); /* Start Sampling */
757  /* Save the real time clock entry for the sample */
758  input->timestamps[input->bufferWriteIdx] = GetLocalTime();
759  } else {
760  /* We entered the ADC_MUXING state */
761  ADS1256_Sync(false); /* TODO: Is this necessary? It may cause temperature fluctuations */
762  }
763 }

771 void ADC_Machine_Reset(void) {
772  /* We use explicit checking here in case other state are added later */
773  if ((CurrentState != ADC_IDLE) && (CurrentState != ADC_INITIALIZED) && (CurrentState != ADC_CHANNEL_SAMPLING)
774  && (CurrentState != ADC_RESET)) {
775 #ifdef ADC_STATE_MACHINE_DEBUG
776  printf("[ADC STATE MACHINE] Attempted to enter ADC_RESET state from %s\n\r", ADCMachine_StringFromState(CurrentState));
777 #endif
778  } else {
779  CurrentState = ADC_RESET;
780  }
781 }
