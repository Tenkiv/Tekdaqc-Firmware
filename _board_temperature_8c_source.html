86 void updateBoardTemperature(Analog_Input_t* input, int32_t code) {
87  int32_t max = MAX_CODE;
88  if (code < 0)
89  ++max; /* Add one for negative range. */
90  temperature = LM35_SLOPE * ((2.0f * V_REFERENCE )/ADS1256_GetGainMultiplier(input->gain))* (((float) code)/max);
91  return;
92 #ifdef BOARD_TEMPERATURE_DEBUG
93  printf("[Board Temperature] New board temperature: %f Deg C.\n\r", temperature);
94 #endif
95  if (max_temp == 0.0f) {
96  EE_ReadVariable(ADDR_BOARD_MAX_TEMP_HIGH, &high_max_word);
97  EE_ReadVariable(ADDR_BOARD_MAX_TEMP_LOW, &low_max_word);
98  scratch = (((uint32_t) high_max_word) << 16) | low_max_word;
99  max_temp = *(float*) (void*)(&scratch);
100  }
101  if (min_temp == 0.0f) {
102  EE_ReadVariable(ADDR_BOARD_MIN_TEMP_HIGH, &high_min_word);
103  EE_ReadVariable(ADDR_BOARD_MIN_TEMP_LOW, &low_min_word);
104  scratch = (((uint32_t) high_min_word) << 16) | low_min_word;
105  min_temp = *(float*) (void*)(&scratch);
106  }
107 
108  if (temperature > max_temp) {
109  printf("[Board Temperature] New board max temperature: %f Deg C.\n\r", temperature);
110  max_temp = temperature;
111  scratch = *(uint32_t*) (void*)(&temperature);
112  EE_WriteVariable(ADDR_BOARD_MAX_TEMP_LOW, (uint16_t) (scratch & 0xFFFF));
113  EE_WriteVariable(ADDR_BOARD_MAX_TEMP_HIGH, (uint16_t) (scratch >> 16));
114  }
115  if (temperature < min_temp) {
116  printf("[Board Temperature] New board min temperature: %f Deg C.\n\r", temperature);
117  min_temp = temperature;
118  scratch = *(uint32_t*) (void*)(&temperature);
119  EE_WriteVariable(ADDR_BOARD_MIN_TEMP_LOW, (uint16_t) (scratch & 0xFFFF));
120  EE_WriteVariable(ADDR_BOARD_MIN_TEMP_HIGH, (uint16_t) (scratch >> 16));
121  }
122 }
