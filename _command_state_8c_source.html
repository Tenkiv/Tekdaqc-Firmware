85 void ServiceTasks(void) {
86  switch (CurrentState) {
87  case UNINITIALIZED:
88  ADC_Machine_Service();
89  DI_Machine_Service();
90  DO_Machine_Service();
91  CurrentState = STATE_IDLE;
92  break;
93  case STATE_IDLE:
94  ADC_Machine_Service();
95  DI_Machine_Service();
96  DO_Machine_Service();
97  break;
98  case STATE_ANALOG_INPUT_SAMPLE:
99  if (ADCSampling == TRUE) {
100  /* Service the ADC state machine */
101  ADC_Machine_Service();
102  } else {
103  CurrentState = STATE_IDLE;
104  }
105  break;
106  case STATE_DIGITAL_INPUT_SAMPLE:
107  if (DISampling == TRUE) {
108  /* Service the DI state machine */
109  DI_Machine_Service();
110  } else {
111  CurrentState = STATE_IDLE;
112  }
113  break;
114  case STATE_DIGITAL_OUTPUT_SAMPLE:
115  if (DOSampling == TRUE) {
116  /* Service the DO state machine */
117  DO_Machine_Service();
118  } else {
119  CurrentState = STATE_IDLE;
120  }
121  break;
122  case STATE_GENERAL_SAMPLE:
123  if ((ADCSampling == TRUE) || (DISampling == TRUE) || (DOSampling == TRUE)) {
124  /* Service the ADC state machine */
125  ADC_Machine_Service();
126  /* Service the DI state machine */
127  DI_Machine_Service();
128  /* Service the DO state machine */
129  DO_Machine_Service();
130  } else {
131  CurrentState = STATE_IDLE;
132  }
133  break;
134  default:
135  /* TODO: Throw error. */
136  break;
137  }
138 }

147 void HaltTasks(void) {
148  printf("[Command State] Halting all tasks.\n\r");
149  switch (CurrentState) {
150  case UNINITIALIZED:
151  /* There is nothing to halt */
152  break;
153  case STATE_IDLE:
154  /* There is nothing to halt */
155  break;
156  case STATE_ANALOG_INPUT_SAMPLE:
157  printf("[Command State] Halting analog input sampling.\n\r");
158  ADC_Machine_Halt();
159  CurrentState = STATE_IDLE;
160  break;
161  case STATE_DIGITAL_INPUT_SAMPLE:
162  printf("[Command State] Halting digital input sampling.\n\r");
163  DI_Machine_Halt();
164  CurrentState = STATE_IDLE;
165  break;
166  case STATE_DIGITAL_OUTPUT_SAMPLE:
167  printf("[Command State] Halting digital output sampling.\n\r");
168  DO_Machine_Halt();
169  CurrentState = STATE_IDLE;
170  break;
171  case STATE_GENERAL_SAMPLE:
172  printf("[Command State] Halting all sampling.\n\r");
173  ADC_Machine_Halt();
174  DI_Machine_Halt();
175  DO_Machine_Halt();
176  CurrentState = STATE_IDLE;
177  break;
178  default:
179  printf("[Command State] Halt tasks called while in an unknown command state.");
180  /* TODO: Throw error */
181  break;
182  }
183 }
