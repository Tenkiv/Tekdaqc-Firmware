83 static void TLE7232_LowLevel_Init(void) {
84  GPIO_InitTypeDef GPIO_InitStructure;
85 
86  /* Enable the SPI clock */
87  TLE7232_SPI_CLK_INIT(TLE7232_SPI_CLK, ENABLE);
88 
89  /* Enable GPIO clocks */
90  RCC_AHB1PeriphClockCmd(TLE7232_SPI_SCK_GPIO_CLK | TLE7232_SPI_MISO_GPIO_CLK |
91  TLE7232_SPI_MOSI_GPIO_CLK | TLE7232_CS_GPIO_CLK, ENABLE);
92 
93  /* SPI pins configuration */
94 
95  /* Connect SPI pins to AF5 */
96  GPIO_PinAFConfig(TLE7232_SPI_SCK_GPIO_PORT, TLE7232_SPI_SCK_SOURCE, TLE7232_SPI_SCK_AF);
97  GPIO_PinAFConfig(TLE7232_SPI_MISO_GPIO_PORT, TLE7232_SPI_MISO_SOURCE, TLE7232_SPI_MISO_AF);
98  GPIO_PinAFConfig(TLE7232_SPI_MOSI_GPIO_PORT, TLE7232_SPI_MOSI_SOURCE, TLE7232_SPI_MOSI_AF);
99 
100  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF;
101  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
102  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
103  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_DOWN;
104 
105  /* SPI SCK pin configuration */
106  GPIO_InitStructure.GPIO_Pin = TLE7232_SPI_SCK_PIN;
107  GPIO_Init(TLE7232_SPI_SCK_GPIO_PORT, &GPIO_InitStructure);
108 
109  /* SPI MOSI pin configuration */
110  GPIO_InitStructure.GPIO_Pin = TLE7232_SPI_MOSI_PIN;
111  GPIO_Init(TLE7232_SPI_MOSI_GPIO_PORT, &GPIO_InitStructure);
112 
113  /* SPI MISO pin configuration */
114  GPIO_InitStructure.GPIO_Pin = TLE7232_SPI_MISO_PIN;
115  GPIO_Init(TLE7232_SPI_MISO_GPIO_PORT, &GPIO_InitStructure);
116 
117  /* Configure TLE7232 CS pin in output pushpull mode */
118  GPIO_InitStructure.GPIO_Pin = TLE7232_CS_PIN;
119  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_OUT;
120  GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
121  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
122  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_NOPULL;
123  GPIO_Init(TLE7232_CS_GPIO_PORT, &GPIO_InitStructure);
124 }

166 static void TLE7232_EvaluateDiagnosis(void) {
167  if (SetFaultStatus != NULL) { /* Don't bother evaluating the faults if we can't do anything */
168  for (uint_fast8_t i = 0; i < NUMBER_TLE7232_CHIPS; ++i) {
169  /* Check the status of this chip */
170  uint16_t mask = 0x0003;
171  uint16_t reg = 0x0000;
172  TLE7232_Status_t status;
173  for (uint16_t shift = 0; shift < 16; shift+=2) { /* The 16bit index is to match the data it operates on */
174  reg = DiagnosisRegisters[i] & (mask << shift); /* Isolate the channel's bits */
175  reg >>= shift; /* Move the bits to the least significant bits */
176  status = (TLE7232_Status_t) reg;
177  if (SetFaultStatus != NULL) {
178  SetFaultStatus(status, i, (shift/2)); /* Set the status on the output */
179  }
180  }
181  }
182  }
183 }

240 uint16_t TLE7232_ReadDiagnosis(uint8_t chip_index) {
241  if ((chip_index >= NUMBER_TLE7232_CHIPS) || (chip_index < 0)) {
242  /* This chip does not exist on the board */
243 #ifdef TLE7232_SPI_DEBUG
244  printf("[TLE7232 SPI] Attempted to write data to a device which does not exist.\n\r");
245 #endif
246  /* TODO: Set an error. */
247  return 0;
248  }
249  /* Select the TLE7232: Chip Select Low */
250  TLE7232_CS_LOW();
251 
252  for (int i = 0; i < NUMBER_TLE7232_CHIPS; ++i) {
253  SPI_I2S_SendData(TLE7232_SPI, TLE7232_CMD_DIAGNOSIS);
254  DiagnosisRegisters[i] = SPI_I2S_ReceiveData(TLE7232_SPI);
255  }
256 
257  while (SPI_I2S_GetFlagStatus(TLE7232_SPI, SPI_I2S_FLAG_BSY) == SET) {
258  /* The SPI transmit buffer has data, so wait patiently */
259  }
260 
261  /* De-select the TLE7232: Chip Select High */
262  TLE7232_CS_HIGH();
263  TLE7232_EvaluateDiagnosis();
264  return DiagnosisRegisters[chip_index];
265 }

382 void TLE7232_WriteRegister(TLE7232_Register_t reg, uint8_t data, uint8_t chip_index) {
383  uint16_t command = TLE7232_CMD_WRITE_REGISTER | reg | data;
384  /* Select the TLE7232: Chip Select Low */
385  TLE7232_CS_LOW();
386 
387  for (int i = 0; i < NUMBER_TLE7232_CHIPS; ++i) {
388  if (i == chip_index) {
389  /* Write the data to the specified chip. */
390  SPI_I2S_SendData(TLE7232_SPI, command);
391  } else {
392  SPI_I2S_SendData(TLE7232_SPI, TLE7232_CMD_DIAGNOSIS);
393  }
394  /* Fetch the diagnosis data regardless of the command */
395  DiagnosisRegisters[i] = SPI_I2S_ReceiveData(TLE7232_SPI);
