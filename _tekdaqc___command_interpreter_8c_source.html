567 static void ProcessCommand(char* command, char raw_args[][MAX_COMMANDPART_LENGTH], uint8_t arg_count) {
568 #ifdef COMMAND_DEBUG
569  printf("[Command Interpreter] Processing command.\n\r");
570 #endif
571  ToUpperCase(command); /* Convert the command string to upper case */
572  Command_t command_type = ParseCommand(command); /* Determine which command this is */
573  char keys[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
574  char values[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
575  ParseKeyValuePairs(raw_args, keys, values, arg_count); /* Parse the argument key/value pairs. */
576  Tekdaqc_Command_Error_t error = ExecuteCommand(command_type, keys, values, arg_count); /* Execute the command */
577  ProcessCommandError(error); /* Handle any errors. */
578 }

586 static void Command_ParseLine(void) {
587 #ifdef COMMAND_DEBUG
588  printf("Parsing command: %s\n\r", interpreter.command_buffer);
589 #endif
590  char raw_args[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
591  char* pch;
592  uint8_t count = 0U;
593  char command[MAX_COMMANDPART_LENGTH];
594  pch = strpbrk(interpreter.command_buffer, COMMAND_DELIMETER);
595  while (pch != NULL ) {
596  ++count;
597  pch = strpbrk(pch + 1, COMMAND_DELIMETER);
598  }
599 #ifdef COMMAND_DEBUG
600  printf("[Command Parser] Parts count: %i\n\r", count);
601 #endif
602  if (count > 0) { /* Command with arguments */
603  /* Extract the command first */
604  int command_end = strcspn(interpreter.command_buffer, COMMAND_DELIMETER);
605  if (command_end == MAX_COMMANDPART_LENGTH) {
606  /* This is not a valid command */
607 #ifdef COMMAND_DEBUG
608  printf("[Command Interpreter] Command was too long, ignoring.\n\r");
609 #endif
610  ClearCommandBuffer();
611  return;
612  } else {
613  /* This is a valid command */
614  pch = strtok(interpreter.command_buffer, COMMAND_DELIMETER);
615  strcpy(command, pch);
616  for (int i = 0; i < count; ++i) {
617  pch = strtok(NULL, COMMAND_DELIMETER);
618  if (pch == NULL ) {
619  break;
620  }
621  strcpy(*(raw_args + i), pch);
622  }
623 #ifdef COMMAND_DEBUG
624  printf("[Command Interpreter] Command: %s\n\r\tParts:\n\r", command);
625  for (int i = 0; i < count; ++i) {
626  printf("\t\tArg[%i]: %s\n\r", i, raw_args[i]);
627  }
628 #endif
629  }
630  } else {
631  /* Command without arguments */
632  int length = strlen(interpreter.command_buffer);
633  if (length > (MAX_COMMANDPART_LENGTH - 1)) {
634  /* This is not a valid command */
635 #ifdef COMMAND_DEBUG
636  printf("[Command Interpreter] Command was too long, ignoring.\n\r");
637 #endif
638  /* Cleanup */
639  ClearCommandBuffer();
640  return;
641  } else {
642  /* Its a valid command */
643  /* Copy the command */
644  strcpy(command, interpreter.command_buffer);
645 #ifdef COMMAND_DEBUG
646  printf("[Command Interpreter] Command: %s\n\r", command);
647 #endif
648  }
649  }
650  /* Process the command */
651  ProcessCommand(command, raw_args, count);
652 }

663 #endif
664  Command_t ret_command = COMMAND_LIST_ANALOG_INPUTS;

669  ++ret_command;

672 #ifdef COMMAND_DEBUG

675  } else {
676  printf("[Command Interpreter] Unable to determine command.\n\r");
677  }
678 #endif
679  return ret_command;

711  break;
712  case ERR_COMMAND_ADC_INVALID_OPERATION:
713  snprintf(TOSTRING_BUFFER, SIZE_TOSTRING_BUFFER, "FAIL - %s", error_string);
714  break;

750 static void ParseKeyValuePairs(char raw_args[][MAX_COMMANDPART_LENGTH], char keys[][MAX_COMMANDPART_LENGTH],
751  char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
752  char* raw;
753  char temp[MAX_COMMANDPART_LENGTH];
754  for (int i = 0; i < count; ++i) {
755  raw = raw_args[i];
756  if (strncmp(raw, KEY_VALUE_PAIR_FLAG, 2) == 0) {
757  /* Properly formatted command */
758  strcpy(temp, (raw + 2)); /* Offset for the first two characters */
759  raw = strtok(temp, KEY_VALUE_PAIR_DELIMETER);
760  ToUpperCase(raw);
761  strcpy(keys[i], raw);
762  raw = strtok(NULL, KEY_VALUE_PAIR_DELIMETER);
763  if (raw != NULL ) {
764  ToUpperCase(raw);
765  strcpy(values[i], raw);
766  } else {
767  strcpy(values[i], "");
768  }
769  } else {
770  /* Not a properly formatted command */
771 #ifdef COMMAND_DEBUG
772  printf("[Command Interpreter] Key/value pair %i (%s) was not properly formatted.\n\r", i, raw);
773 #endif
774  }
775  }
776 }

787 static bool InputArgsCheck(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count, uint8_t num_params,
788  const char** params) {
789  if (count > num_params) {
790  return false;
791  }
792  int idx = 0;
793  bool valid = false;
794  for (int i = 0; i < count; ++i) {
795  idx = 0;
796  valid = false;
797  while (idx < num_params) {
798  if (strcmp(keys[i], params[idx++]) == 0) {
799  valid = true;
800  break;
801  }
802  }
803  if (valid == false) {
804  /* This is a bad key */
805  return false;
806  }
807  }
808  return true;
809 }

836 static void BuildAnalogInputList(Channel_List_t list_type, char* param) {
837  uint8_t count = 0U;
838  char* ptr = NULL;
839  int8_t channel = -1;
840  long int value = 0L;
841  char* str = NULL;
842  uint8_t start = 0U;
843  uint8_t end = 0U;
844  uint8_t value1 = 0U;
845  uint8_t value2 = 0U;
846 
847  for (uint_fast8_t i = 0; i < NUM_ANALOG_INPUTS; ++i) {
848  aInputs[i] = NULL; /* NULL them all initially */
849  }
850 
851  switch (list_type) {
852  case SINGLE_CHANNEL: /* SINGLE_CHANNEL */
853  channel = (int8_t) strtol(param, NULL, 10);
854  if (channel < 0 || channel > NUM_ANALOG_INPUTS) {
855  /* Input number out of range */
856 #ifdef COMMAND_DEBUG
857  printf("[Command Interpreter] The requested input number is out of range.\n\r");
858 #endif
859  break;
860  }
861  aInputs[0] = GetAnalogInputByNumber(channel);
862  break;
863  case CHANNEL_SET: /* CHANNEL_SET */
864  value = strtol(param, &ptr, 10);
865  while (value != 0L) {
866  ++count;
867  if (*ptr == SET_DELIMETER) {
868  str = ptr + 1;
869  }
870  value = strtol(str, &ptr, 10);
871  }
872  value = strtol(param, &ptr, 10);
873  for (int i = 0; i < count; ++i) {
874  if (*ptr == SET_DELIMETER) {
875  str = ptr + 1;
876  }
877  aInputs[i] = GetAnalogInputByNumber(strtol(str, &ptr, 10));
878  }
879  break;
880  case CHANNEL_RANGE: /* INPUT_RANGE */
881  start = 0U;
882  end = 0U;
883  value1 = (uint8_t) strtol(param, &ptr, 10); /* We know these can potentially loose data...it would be invalid anyway */
884  value2 = (uint8_t) strtol((ptr + 1), NULL, 10);
885  if (value1 != 0L) {
886  start = value1;
887  if (value2 != 0L) {
888  end = value2;
889  } else {
890  end = NUM_ANALOG_INPUTS;
891  }
892  } else {
893  start = 0U;
894  }
895  count = end - start + 1;
896  for (int i = 0; i < count; ++i) {
897  aInputs[i] = GetAnalogInputByNumber(start + i); /* Some of these may be NULL, this is OK. */
898  }
899  break;
900  case ALL_CHANNELS: /* ALL_CHANNELS */
901  count = NUM_ANALOG_INPUTS;
902  for (int i = 0; i < count; ++i) {
903  aInputs[i] = GetAnalogInputByNumber(i); /* Some of these may be NULL, this is OK. */
904  }
905  break;
906  default:
907 #ifdef COMMAND_DEBUG
908  printf("[Command Interpreter] The specified input range is invalid.\n\r");
