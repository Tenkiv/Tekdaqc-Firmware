588 static void ProcessCommand(char* command, char raw_args[][MAX_COMMANDPART_LENGTH], uint8_t arg_count) {
589 #ifdef COMMAND_DEBUG
590  printf("[Command Interpreter] Processing command.\n\r");
591 #endif
592  ToUpperCase(command); /* Convert the command string to upper case */
593  Command_t command_type = ParseCommand(command); /* Determine which command this is */
594  char keys[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
595  char values[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
596  Tekdaqc_Command_Error_t error = ERR_COMMAND_OK;
597  bool success = ParseKeyValuePairs(raw_args, keys, values, arg_count); /* Parse the argument key/value pairs. */
598  if (success == TRUE) {
599  error = ExecuteCommand(command_type, keys, values, arg_count); /* Execute the command */
600  } else {
601  error = ERR_COMMAND_PARSE_ERROR;
602  }
603  ProcessCommandError(error); /* Handle any errors. */
604 }

612 static void Command_ParseLine(void) {
613 #ifdef COMMAND_DEBUG
614  printf("Parsing command: %s\n\r", interpreter.command_buffer);
615 #endif
616  char raw_args[MAX_NUM_ARGUMENTS][MAX_COMMANDPART_LENGTH];
617  char* pch;
618  uint8_t count = 0U;
619  char command[MAX_COMMANDPART_LENGTH];
620  pch = strpbrk(interpreter.command_buffer, COMMAND_DELIMETER);
621  while (pch != NULL) {
622  ++count;
623  pch = strpbrk(pch + 1, COMMAND_DELIMETER);
624  }
625 #ifdef COMMAND_DEBUG
626  printf("[Command Parser] Parts count: %i\n\r", count);
627 #endif
628  if (count > 0) { /* Command with arguments */
629  /* Extract the command first */
630  int command_end = strcspn(interpreter.command_buffer, COMMAND_DELIMETER);
631  if (command_end == MAX_COMMANDPART_LENGTH) {
632  /* This is not a valid command */
633 #ifdef COMMAND_DEBUG
634  printf("[Command Interpreter] Command was too long, ignoring.\n\r");
635 #endif
636  ClearCommandBuffer();
637  return;
638  } else {
639  /* This is a valid command */
640  pch = strtok(interpreter.command_buffer, COMMAND_DELIMETER);
641  strcpy(command, pch);
642  for (int i = 0; i < count; ++i) {
643  pch = strtok(NULL, COMMAND_DELIMETER);
644  if (pch == NULL) {
645  break;
646  }
647  strcpy(*(raw_args + i), pch);
648  }
649 #ifdef COMMAND_DEBUG
650  printf("[Command Interpreter] Command: %s\n\r\tParts:\n\r", command);
651  for (int i = 0; i < count; ++i) {
652  printf("\t\tArg[%i]: %s\n\r", i, raw_args[i]);
653  }
654 #endif
655  }
656  } else {
657  /* Command without arguments */
658  int length = strlen(interpreter.command_buffer);
659  if (length > (MAX_COMMANDPART_LENGTH - 1)) {
660  /* This is not a valid command */
661 #ifdef COMMAND_DEBUG
662  printf("[Command Interpreter] Command was too long, ignoring.\n\r");
663 #endif
664  /* Cleanup */
665  ClearCommandBuffer();
666  return;
667  } else {
668  /* Its a valid command */
669  /* Copy the command */
670  strcpy(command, interpreter.command_buffer);
671 #ifdef COMMAND_DEBUG
672  printf("[Command Interpreter] Command: %s\n\r", command);
673 #endif
674  }
675  }
676  /* Process the command */
677  ProcessCommand(command, raw_args, count);
678 }

686 static Command_t ParseCommand(const char* command) {
687 #ifdef COMMAND_DEBUG
688  printf("[Command Interpreter] Parsing command.\n\r");
689 #endif
690  Command_t ret_command = COMMAND_LIST_ANALOG_INPUTS;
691  while (ret_command <= COMMAND_NONE) {
692  if (strcmp(command, COMMAND_STRINGS[ret_command]) == 0) {
693  break;
694  } else {
695  ++ret_command;
696  }
697  }
698 #ifdef COMMAND_DEBUG
699  if (ret_command <= COMMAND_NONE) {
700  printf("[Command Interpreter] Determined command to be: %s\n\r", COMMAND_STRINGS[ret_command]);
701  } else {
702  printf("[Command Interpreter] Unable to determine command.\n\r");
703  }
704 #endif
705  return ret_command;
706 }

772 static bool ParseKeyValuePairs(char raw_args[][MAX_COMMANDPART_LENGTH], char keys[][MAX_COMMANDPART_LENGTH],
773  char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
774  char* raw;
775  char temp[MAX_COMMANDPART_LENGTH];
776  bool retval = TRUE;
777  for (int i = 0; i < count; ++i) {
778  raw = raw_args[i];
779  if (strncmp(raw, KEY_VALUE_PAIR_FLAG, 2) == 0) {
780  /* Properly formatted command */
781  strcpy(temp, (raw + 2)); /* Offset for the first two characters */
782  raw = strtok(temp, KEY_VALUE_PAIR_DELIMETER);
783  ToUpperCase(raw);
784  strcpy(keys[i], raw);
785  raw = strtok(NULL, KEY_VALUE_PAIR_DELIMETER);
786  if (raw != NULL) {
787  ToUpperCase(raw);
788  strcpy(values[i], raw);
789  } else {
790  strcpy(values[i], "");
791  }
792  } else {
793  /* Not a properly formatted command */
794 #ifdef COMMAND_DEBUG
795  printf("[Command Interpreter] Key/value pair %i (%s) was not properly formatted.\n\r", i, raw);
796 #endif
797  retval = FALSE;
798  break;
799  }
800  }
801  return retval;
802 }

813 static bool InputArgsCheck(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count,
814  uint8_t num_params, const char** params) {
815  if (count > num_params) {
816  return false;
817  }
818  int idx = 0;
819  bool valid = false;
820  for (int i = 0; i < count; ++i) {
821  idx = 0;
822  valid = false;
823  while (idx < num_params) {
824  if (strcmp(keys[i], params[idx++]) == 0) {
825  valid = true;
826  break;
827  }
828  }
829  if (valid == false) {
830  /* This is a bad key */
831  return false;
832  }
833  }
834  return true;
835 }

862 static void BuildAnalogInputList(Channel_List_t list_type, char* param) {
863  uint8_t count = 0U;
864  char* ptr = NULL;
865  int8_t channel = -1;
866  long int value = 0L;
867  char* str = NULL;
868  uint8_t start = 0U;
869  uint8_t end = 0U;
870  uint8_t value1 = 0U;
871  uint8_t value2 = 0U;
872 
873  for (uint_fast8_t i = 0; i < NUM_ANALOG_INPUTS; ++i) {
874  aInputs[i] = NULL; /* NULL them all initially */
875  }
876 
877  switch (list_type) {
878  case SINGLE_CHANNEL: /* SINGLE_CHANNEL */
879  channel = (int8_t) strtol(param, NULL, 10);
880  if (channel < 0 || channel > NUM_ANALOG_INPUTS) {
881  /* Input number out of range */
882 #ifdef COMMAND_DEBUG
883  printf("[Command Interpreter] The requested input number is out of range.\n\r");
884 #endif
885  break;
886  }
887  aInputs[0] = GetAnalogInputByNumber(channel);
888  break; /* END SINGLE_INPUT */
889  case CHANNEL_SET: /* CHANNEL_SET */
890  if (param != NULL) {
891  str = param;
892  while (TRUE) {
893  if (*ptr == SET_DELIMETER) {
894  str = ptr + 1;
895  }
896  value = strtol(str, &ptr, 10);
897  if (value < NUM_ANALOG_INPUTS && value >= 0L) {
898  aInputs[value] = GetAnalogInputByNumber(value);
899  ++count;
900  }
901  if (ptr == NULL) {
902  break;
903  }
904  }
905  }
906  break; /* END INPUT SET */
907  case CHANNEL_RANGE: /* INPUT_RANGE */
908  start = 0U;
909  end = 0U;
910  value1 = (uint8_t) strtol(param, &ptr, 10); /* We know these can potentially loose data...it would be invalid anyway */
911  value2 = (uint8_t) strtol((ptr + 1), NULL, 10);
912 
913  if (value1 != 0L) {
914  start = value1;
915  } else {
916  start = 0U;
917  }
918  if (value2 != 0L) {
919  end = value2;
920  } else {
921  end = NUM_ANALOG_INPUTS;
922  }
923 
924  count = end - start + 1;
925  for (int i = 0; i < count; ++i) {
926  aInputs[i] = GetAnalogInputByNumber(start + i); /* Some of these may be NULL, this is OK. */
927  }
928  break; /* END INPUT RANGE */
929  case ALL_CHANNELS: /* ALL_CHANNELS */
