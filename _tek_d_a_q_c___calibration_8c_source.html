74 static TekDAQC_Function_Error_t SetADCParameters(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
75  TekDAQC_Function_Error_t retval = ERR_FUNCTION_OK;
76  int8_t index = -1;
77  ADS1256_PGA_t pga = ADS1256_PGAx1;
78  ADS1256_SPS_t rate = ADS1256_SPS_60;
79  ADS1256_BUFFER_t buffer = ADS1256_BUFFER_DISABLED;
80  for (uint_fast8_t i = 0; i < NUM_SYSTEM_CAL_PARAMS; ++i) {
81  index = GetIndexOfArgument(keys, SYSTEM_CAL_PARAMS[i], count);
82  if (index >= 0) { /* We found the key in the list */
83  switch (i) { /* Switch on the key not position in arguments list */
84  case 0: /* BUFFER key */
85  buffer = ADS1256_StringToBuffer(values[index]);
86  break;
87  case 1: /* RATE key */
88  rate = ADS1256_StringToDataRate(values[index]); /* We use the discovered index for this key */
89  break;
90  case 2: /* GAIN key */
91  pga = ADS1256_StringToPGA(values[index]); /* We use the discovered index for this key */
92  break;
93  default:
94  retval = ERR_AIN_PARSE_ERROR;
95  }
96  } else if (i == 0 || i == 1 || i == 2 || i == 3) {
97  /* The GAIN, RATE and BUFFER keys are not strictly required */
98  continue;
99  } else {
100  /* Somehow an error happened */
101 #ifdef CALIBRATION_DEBUG
102  printf("[Command Interpreter] Unable to locate key: %s\n\r", READ_ANALOG_KEYS[i]);
103 #endif
104  retval = ERR_AIN_PARSE_MISSING_KEY; /* Failed to locate a key */
105  }
106  }
107  ADS1256_SetInputBufferSetting(buffer);
108  ADS1256_SetDataRate(rate);
109  ADS1256_SetPGASetting(pga);
110  return retval;
111 }

139 TekDAQC_Function_Error_t PerformSystemGainCalibration(char keys[][MAX_COMMANDPART_LENGTH], char values[][MAX_COMMANDPART_LENGTH], uint8_t count) {
140  TekDAQC_Function_Error_t retval = ERR_FUNCTION_OK;
141  retval = SetADCParameters(keys, values, count);
142  PhysicalAnalogInput_t input = EXTERNAL_0;
143  int8_t index = -1;
144  for (uint_fast8_t i = 0; i < NUM_SYSTEM_GCAL_PARAMS; ++i) {
145  index = GetIndexOfArgument(keys, SYSTEM_GCAL_PARAMS[i], count);
146  if (index >= 0) { /* We found the key in the list */
147  switch (i) { /* Switch on the key not position in arguments list */
148  case 3: /* INPUT key */
149  input = ADS1256_StringToBuffer(values[index]);
150  break;
151  default:
152  retval = ERR_AIN_PARSE_ERROR;
153  }
154  } else if (i == 0 || i == 1 || i == 2) {
155  /* The GAIN, RATE and BUFFER keys are not strictly required */
156  continue;
157  } else {
158  /* Somehow an error happened */
159 #ifdef CALIBRATION_DEBUG
160  printf("[Command Interpreter] Unable to locate key: %s\n\r", READ_ANALOG_KEYS[i]);
161 #endif
162  retval = ERR_AIN_PARSE_MISSING_KEY; /* Failed to locate a key */
163  }
164  }
165  ADC_GainCalibrate(input);
166  return retval;
167 }
